52 void
53 AL3DFile::load(QFile * file)
54 {
55  //Reset object.
56  clear();
57 
58  //Grab type string to make sure it is an al3d file.
59  char type[17];
60  qint64 bytesRead = file->read(type, 17);
61  if (17 != bytesRead) //Fatal error.
62  {
63  qDebug() << file->fileName() <<
64  ": Error reading file identification header.";
65  return;
66  }
67  if (! strncmp(type, "AliconaImaging\r\n", 17)) {
68  // Wrong file type
69  qDebug() << file->fileName() <<
70  " is not an .al3D file.";
71  return;
72  }
73 
74  //Read tags.
75  //Read the first two tags to get the Version
76  //and number of tags.
77  bool success;
78  for (int i = 0; i < 2; ++i)
79  success = read_tag(file);
80 
81  if (!success) //Fatal error.
82  {
83  qDebug() << file->fileName() <<
84  ": Error reading file identification header.";
85  return;
86  }
87 
88  int tag_count = get_tag("TagCount").toInt();
89 
90  for (int i = 0; i < tag_count; ++i) {
91  success = read_tag(file);
92  }
93 
94  if (!success)
95  {
96  qDebug() << file->fileName() <<
97  ": File is missing information.";
98  }
99 
100  //Get the user-defined comment.
101  comment = QString(file->read(256));
102 
103  //Get the icon if there is one.
104  QVariant icon_offset = get_tag("IconOffset");
105  if ((! icon_offset.isNull()) && (icon_offset.toLongLong() != 0)
106  && (file->pos() <= icon_offset.toLongLong())) {
107 
108  file->seek(icon_offset.toLongLong());
109  success = read_icon(file);
110  if (!success)
111  {
112  qDebug() << file->fileName() <<
113  ": Icon failed to read properly.";
114  icon = QImage();
115  }
116  }
117 
118  //Retrieve the size of the data.
119  QVariant cols = get_tag("Cols");
120  QVariant rows = get_tag("Rows");
121  image_size = QSize(cols.toInt(), rows.toInt());
122  if ((cols.isNull()) || (rows.isNull()) ||
123  (image_size.width() <= 0) || (image_size.height() <= 0))
124  {
125  qDebug() << file->fileName()
126  << ": Invalid data width or height.";
127  return;
128  }
129 
130  //Retrieve the pixel size.
131  QVariant pixX = get_tag("PixelSizeXMeter");
132  QVariant pixY = get_tag("PixelSizeYMeter");
133  pixelSizeX = pixX.toFloat();
134  pixelSizeY = pixY.toFloat();
135  if ((pixX.isNull()) || (pixY.isNull()) ||
136  (pixelSizeX < FLT_MIN) || (pixelSizeY < FLT_MIN))
137  {
138  qDebug() << file->fileName()
139  << ": Invalid pixel sizes.";
140  return;
141  }
142 
143  //Get the depth data and mask.
144  QVariant depth_data_offset = get_tag("DepthImageOffset");
145  if ((! depth_data_offset.isNull()) && (depth_data_offset.toLongLong() != 0)) {
146  file->seek(depth_data_offset.toLongLong());
147  success = read_depth_data(file);
148  if (!success) //fatal error
149  {
150  qDebug() << file->fileName() <<
151  ": Depth data failed to read properly.";
152  delete [] depth_data;
153  depth_data = NULL;
154  }
155  else
156  {
157  QVariant maskedValue = get_tag("InvalidPixelValue");
158  if (!maskedValue.isNull())
159  {
160  //Populate mask.
161  //Init mask to size of image and fill with false.
162  int numel = image_size.width()
163  *image_size.height();
164  mask = QBitArray(numel);
165  //Set valid data points to true
166  for(int i = 0; i < numel; ++i)
167  {
168  if (maskedValue.toFloat() !=
169  depth_data[i])
170  {
171  mask.setBit(i);
172  }
173  }
174  }
175  }
176  }
177 
178  //Get the texture data.
179  QVariant texture_offset = get_tag("TextureImageOffset");
180  QVariant planes = get_tag("NumberOfPlanes");
181  if ((!texture_offset.isNull()) && (texture_offset.toLongLong() != 0)
182  && (!planes.isNull()) && (planes.toInt() > 0)) {
183  file->seek(texture_offset.toLongLong());
184  for (int i = 0; i < planes.toInt(); ++i) {
185  success = read_image_plane(file);
186  if (!success)
187  {
188  qDebug() << file->fileName() <<
189  ": Texture data plane " << i
190  << " failed to read properly";
191  }
192  }
193  }
194 
195  //Successful read.
196  null = false;
197 }

219 QImage
220 AL3DFile::get_image_by_tag(const QString& tag)
221 {
222  //Check tag
223  QVariant tag_val = get_tag(tag);
224  // If we don't have that tag, return a null image
225  if (tag_val.isNull())
226  return QImage();
227  QString ptr_str = tag_val.toString();
228  // If we have an empty tag, return a null image
229  if (ptr_str.isEmpty())
230  return QImage();
231 
232  //Valid tag, split it.
233  QStringList strings = ptr_str.split(";");
234  //Do what the tag says
235  if (strings.size() == 0) { //If no planes
236  return QImage(); //return null image
237  }
238  else if (strings.size() == 1) { //If one plane
239  //Return grayscale image.
240  // TODO: Check for valid int
241  return image_planes.at(strings.at(0).toInt());
242  }
243  else if (strings.size() == 3) { //If 3 planes
244 
245  //Assemble an RGB image
246  int planes[3];
247  for (int i = 0; i < 3; ++i) {
248  planes[i] = strings.at(i).toInt();
249  // TODO: Check for valid int
250  }
251 
252  // Create a new image
253  QImage image(image_size, QImage::Format_RGB888);
254  uchar * image_data = image.bits();
255 
256  // Interlace the image data
257  int width = image_size.width();
258  int height = image_size.height();
259  int stride = image.bytesPerLine();
260  for (int p = 0; p < 3; ++p) {
261  QImage plane = image_planes.at(planes[p]);
262  int plane_stride = plane.bytesPerLine();
263  const uchar * plane_data = plane.bits();
264 
265  for (int r = 0; r < height; ++r) {
266  for (int c = 0; c < width; ++c) {
267  image_data[r * stride + 3*c + p] =
268  plane_data[r * plane_stride + c];
269  }
270  }
271  }
272 
273  return image;
274  }
275  else { //Something is very wrong.
276  qDebug() << tag <<
277  " has an invalid number of image planes.\n"
278  << "Returning null QImage.";
279  return QImage();
280  }
281 }

360 bool
361 AL3DFile::read_icon(QFile * file)
362 {
363  int iconSize = 150;
364  int iconSize8 = 152; //152 for 8-byte alignment.
365 
366  icon = QImage(QSize(iconSize, iconSize), QImage::Format_RGB888);
367  uchar * icon_data = icon.bits();
368  int stride = icon.bytesPerLine();
369  for (int p = 0; p < 3; ++p) { //3 planes.
370  for (int r = 0; r < iconSize; ++r) { //number of rows
371  // Read a row
372  char* row = new char[iconSize8];
373  qint64 bytesRead = file->read(row, iconSize8);
374  if (bytesRead != iconSize8)
375  {
376  delete [] row;
377  return false; //failure to properly read icon
378  }
379  // Fill the row
380  for (int c = 0; c < iconSize; ++c) { //150 cols.
381  icon_data[r * stride + (3*c + p)] = row[c];
382  }
383  delete [] row;
384  }
385  }
386 
387  return true; //success
388 }

390 bool
391 AL3DFile::read_depth_data(QFile * file)
392 {
393  //Make a datastream to ensure correct endianness and precision.
394  QDataStream stream (file);
395  stream.setByteOrder(QDataStream::LittleEndian);
396  stream.setFloatingPointPrecision(
397  QDataStream::SinglePrecision);
398 
399  // Cache these
400  int width = image_size.width();
401  int height = image_size.height();
402 
403  //8 byte alignment.
404  int line_width = width;
405  // Padd to 8 bytes
406  if (line_width % 2) //If odd
407  ++line_width; //Make it even.
408 
409  // re-allocate the depth data
410  depth_data = new float[width * height];
411 
412  //Read in the depth data.
413  for (int i = 0; i < height; ++i)
414  {
415  for (int j = 0; j < line_width; ++j)
416  {
417  if (j < width)
418  {
419  stream >> depth_data[i*width + j];
420  }
421  else
422  {
423  float junk;
424  stream >> junk;
425  }
426  }
427  }
428 
429  if (stream.status() != QDataStream::Ok)
430  {
431  qDebug() << file->fileName()
432  << ": Either read past end or"
433  << " read corrupt data.";
434  return false;
435  }
436  else
437  {
438  return true; //success
439  }
440 }

443 AL3DFile::read_image_plane(QFile * file)
444 {
445  // Cache these
446  int width = image_size.width();
447  int height = image_size.height();
448 
449  int line_width = width;
450  // Pad to 8 bytes
451  if (line_width % 8)
452  line_width = line_width + (8 - (line_width % 8));
453 
454  // Create a new QImage
455  QImage image(image_size, QImage::Format_Indexed8);
456  image.setColorCount(256);
457  // Set the color table
458  for (int i = 0; i < 256; ++i)
459  image.setColor(i, qRgb(i, i, i));
460  //Get pointer to image data in QImage.
461  uchar * image_data = image.bits();
462  int stride = image.bytesPerLine();
463 
464  // Create storage space
465  char * buffer = new char[line_width];
466 
467  for (int r = 0; r < height; ++r) {
468  qint64 bytesRead = file->read(buffer, line_width);
469  if (line_width != bytesRead)
470  {
471  return false; //problem reading data.
472  }
473 
474  // I could use memcpy here, but I'm not sure what it would do with
475  // alignment, so I'll leave it alone.
476  memcpy(image_data + (r * stride), buffer, line_width);
477  }
478 
479  delete [] buffer;
480 
481  // Add it to the list
482  image_planes.append(image);
483 
484  return true;
485 }
