44 class IntNonrigidCorSampWithFlips {
45 public:
46  std::auto_ptr<std::vector<double> >
47  operator()(const std::vector<int>& y1,
48  const std::vector<int>& y2,
49  size_t pairs,
50  size_t window)
51  {
52  std::auto_ptr<std::vector<double> >
53  result(new std::vector<double>(pairs));
54  size_t r = y1.size() - window + 1;
55 
56  for (size_t j = 0; j < pairs; ++j) {
57  std::vector<int>::const_iterator y1Begin =
58  y1.begin() + static_cast<size_t>(r*random01());
59 
60  //An offset into y2.
61  size_t i2 = static_cast<size_t>(r*random01());
62 
63  if (random01() < 0.5) {
64  (*result)[j] = intCompCorr(y1Begin, y2.begin() + i2, window);
65  }
66  else {
67  //y2.rbegin() returns a reverse iterator pointing
68  //to the last element in y2. Adding i2 to it
69  //makes it point to an element i2 elements from the
70  //end. And as a reverse iterator, the other operations
71  //(++) will also act in reverse, hence the usefulness
72  //of this type of iterator here.
73  (*result)[j] = intCompCorr(y1Begin, y2.rbegin() + i2, window);
74  }
75  }
76  return result;
77  }
78 };

80 class IntRigidCorSampExcludeSearch {
81  public:
90  std::auto_ptr<std::vector<double> >
91  operator()(const std::vector<int>& y1,
92  const std::vector<int>& y2,
93  int l1,
94  int l2,
95  size_t searchWindow,
96  size_t pairs,
97  size_t window)
98  {
99  std::auto_ptr<std::vector<double> >
100  result(new std::vector<double>(pairs));
101  size_t nn1 = y1.size();
102  size_t nn2 = y2.size();
103 
104  //i is a random offset from both locations, l1 and l2.
105  //We need a window with lower index at offset i to
106  //lie totally within each sample, y1 and y2.
107  //Then i + min(l1, l2) >= 0 and so i >= -min(l1, l2).
108  //And we need i + max(l1, l2) <= nn - window and so
109  //i <= nn - window - max(l1, l2). (To see that nn - window
110  //is the max allowable lower index, just consider
111  //if nn = 3 and window = 3. Clearly, then only the first
112  //index, 0, is allowable.)
113 
114  // added by maverick:
115  // since the lengths for the two traces might be different,
116  // some adjustment is added here.
117  int ilower = -std::min(l1, l2);
118  int iupper = std::min(nn1 - l1, nn2 - l2) - window;
119  //int iupper = nn - std::max(l1, l2) - window;
120 
121  int sw = static_cast<int>(searchWindow);
122  int w = static_cast<int>(window);
123 
124  //Ru He comments:
125  //in order to get the rigid window on the left side of both search window, ilower <= -w, i.e., -ilower >= w
126  //in order to get the rigid window on the right side of both search window, iupper >= sw
127  if (ilower > -w && iupper < sw) {
128  (*result).resize(0);
129  return result;
130  }
131 
132  // somehow we do not want to get a rigid pair in the windows where
133  // we have the maximum, but maverick does not know
134  // the specific ristrictiion.
135  // added by maverick
136  RandomInSplitRange splitRand(ilower, -w, sw, iupper);
137 
138  std::vector<int>::const_iterator y1Begin = y1.begin();
139  std::vector<int>::const_iterator y2Begin = y2.begin();
140  for (size_t j = 0; j < pairs; ++j) {
141  //Ru He comments: i will take range from [ilower, -w] U [sw, iupper]
142  int i = splitRand();
143  (*result)[j] = intCompCorr(y1Begin + l1 + i, y2Begin + l2 + i, window);
144  }
145  return result;
146  }
147 };

188  //leftShiftLBTrace1 is the left shift lower bound from l1

194  //rightShiftUBTrace2 is the right shift upper bound from l2
195  int leftShiftLBTrace2 = -l2;
196  int rightShiftUBTrace2 = nn2 - l2 - randomWindow;
197 
198  //Ru He comments:
199  //in order to get the random window on the left side of the search window in Trace1, leftShiftLBTrace1 <= leftShiftUB ;
200  //in order to get the random window on the right side of the search window in Trace1, rightShiftLB <= rightShiftUBTrace1;
201  //in order to get the random window on the left side of the search window in Trace2, leftShiftLBTrace2 <= leftShiftUB ;
202  //in order to get the random window on the right side of the search window in Trace2, rightShiftLB <= rightShiftUBTrace2;
203 
204  //if random window can not be sampled from Trace 1 or random window can not be sampled from Trace 2, then return empty vector
205  if((leftShiftLBTrace1 > leftShiftUB && rightShiftLB > rightShiftUBTrace1)
206  || (leftShiftLBTrace2 > leftShiftUB && rightShiftLB > rightShiftUBTrace2 )){
207  (*result).resize(0);
208  return result;
209  }
210 
211  RandomInSplitRange splitRandTrace1(leftShiftLBTrace1, leftShiftUB, rightShiftLB, rightShiftUBTrace1);
212  RandomInSplitRange splitRandTrace2(leftShiftLBTrace2, leftShiftUB, rightShiftLB, rightShiftUBTrace2);
213 
214  std::vector<int>::const_iterator y1Begin = y1.begin();
215  std::vector<int>::const_iterator y2Begin = y2.begin();
216 
217  //Ru He comments:

222  //Ru He comments:
223  //shift1 will take range from [leftShiftLBTrace1, leftShiftUB] U [rightShiftLB, rightShiftUBTrace1]
224  //shift2 will take range from [leftShiftLBTrace2, leftShiftUB] U [rightShiftLB, rightShiftUBTrace2]
225  shift1 = splitRandTrace1();

228  }
