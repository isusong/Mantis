88 QPoint
89 CleanVirtualMark::findMarkEdges(const QVector<float> data)
90 {
91  //Approximate the derivative with partial differences.
92  QVector<float> diff;
93  for (int i = 1; i < data.size(); ++i)
94  {
95  diff.push_back(data[i] - data[i-1]);
96  }
97 
98  //Filter the derivative to remove gross noise.
99  QVector<float> diffCopy (diff);
100  CFilterImage filter(diff.size(), 1); //1D derivative.
101  filter.GaussianFilterFloat(diff.data(), diffCopy.data(), 15);
102 
103  //Convert the filtered derivative to a vector of doubles
104  //for the statistics calculation.
105  QVector<double> diffDouble;
106  for (int i = 0; i < diffCopy.size(); ++i)
107  diffDouble.push_back((double) diffCopy[i]);
108  //Don't need these anymore.
109  diff.clear();
110  diffCopy.clear();
111 
112  //Locate the edges of the mark using an iterative approach.
113  int border = 20; //Minimum to get cut off on each end.
114  int startIndex = 0;
115  int endIndex = diffDouble.size() - 1;
116  for (int j = 0; j < 2; ++j) //Repeat
117  {
118  //Compute the mean and standard deviation of the filtered
119  //derivative using stats.h in the statistics library:
120  double mean, stddev;
121  //Operate on trimmed copy.
122  QVector<double> diffDoubleTrimmed;
123  for (int i = startIndex; i < endIndex + 1; ++i)
124  diffDoubleTrimmed.push_back(diffDouble[i]);
125  meanAndVar(diffDoubleTrimmed.toStdVector(), mean, stddev);
126  stddev = sqrt(stddev); //Variance -> stddev.
127 
128  //Look for ends where the absolute value of (the
129  //derivative - mean) is greater than stddev.
130  bool startFound = false;
131  bool endFound = false;
132  int oldStartIndex = startIndex;
133  int oldEndIndex = endIndex;
134  int diffSize = diffDouble.size();
135  for (int i = 0; i < diffSize; ++i)
136  {
137  if (!startFound &&
138  (abs(diffDouble[i] - mean) > stddev))
139  {
140  startIndex = i + 1;
141  }
142  else if (i > border)
143  startFound = true;
144 
145  if (!endFound &&
146  (abs(diffDouble[diffSize-1-i] - mean) > stddev))
147  {
148  endIndex = diffSize - 1 - i - 1;
149  }
150  else if (i > border)
151  endFound = true;
152  }
153 
154  //If the new start and end points were
155  //never found, return original start and end points.
156  if (!startFound || !endFound) //something's wrong
157  {
158  int startIndex = oldStartIndex;
159  int endIndex = oldEndIndex;
160  break; //don't repeat.
161  }
162 
163  }
164 
165  return QPoint(startIndex, endIndex);
166 }
