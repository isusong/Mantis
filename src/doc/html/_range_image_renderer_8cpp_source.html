81 void
82 RangeImageRenderer::computeBoundingBox()
83 {
84  float minX = FLT_MAX;
85  float maxX = -FLT_MAX;
86  float minY = FLT_MAX;
87  float maxY = -FLT_MAX;
88  float minZ = FLT_MAX;
89  float maxZ = -FLT_MAX;
90 
91  for (int i = 0; i < width*height; ++i)
92  {
93  if (mask.testBit(i))
94  {
95  float currentX = (i%width)*pixelSizeX;
96  float currentY = (i/width)*pixelSizeY;
97  float currentZ = depth[i];
98  if (currentX < minX)
99  {
100  minX = currentX;
101  }
102  if (currentX > maxX)
103  {
104  maxX = currentX;
105  }
106  if (currentY < minY)
107  {
108  minY = currentY;
109  }
110  if (currentY > maxY)
111  {
112  maxY = currentY;
113  }
114  if (currentZ < minZ)
115  {
116  minZ = currentZ;
117  }
118  if (currentZ > maxZ)
119  {
120  maxZ = currentZ;
121  }
122  }
123  }
124  bbMin = QVector3D(minX, minY, minZ);
125  bbMax = QVector3D(maxX, maxY, maxZ);
126 }

227 void
228 RangeImageRenderer::initBuffers()
229 {
230  //Downsampled width & height.
231  int dWidth = width/skip;
232  int dHeight = height/skip;
233 
234  //Populate vertices & texture coordinates.
235  GLfloat* xyz =
236  new GLfloat[3*dWidth*dHeight];
237  GLfloat* texCoords = new GLfloat[2*dWidth*dHeight];
238  for(int i = 0; i < dHeight; ++i)
239  {
240  for(int j = 0; j < dWidth; ++j)
241  {
242  int idx = dWidth*i + j;
243  xyz[3*idx + 0] = (GLfloat) j*skip*pixelSizeX;
244  xyz[3*idx + 1] = (GLfloat) i*skip*pixelSizeY;
245  xyz[3*idx + 2] = (GLfloat)
246  depth[width*i*skip + j*skip];
247  texCoords[2*idx] = (1/((float) dWidth)) * (j + 0.5f);
248  texCoords[2*idx + 1] = (1/((float) dHeight))
249  * (i + 0.5f);
250  }
251  }
252  //Init VBO
253  vbo.create();
254  vbo.bind();
255  vbo.setUsagePattern(QGLBuffer::StaticDraw);
256  vbo.allocate(xyz, 3*dWidth*dHeight*sizeof(GLfloat));
257  //Init TBO
258  tbo.create();
259  tbo.bind();
260  tbo.setUsagePattern(QGLBuffer::StaticDraw);
261  tbo.allocate(texCoords, sizeof(GLfloat)*2*dWidth*dHeight);
262  delete [] texCoords;
263 
264  //Do meshing (populate indices and normals).
265  QVector<GLuint> indices;
266  QVector<GLfloat> normals (3*dWidth*dHeight, 0);
267  //Generate mesh
268  //Go through the spaces between the rows.
269  for (int i = 0; i < dHeight - 1; ++i)
270  {
271  //Go through the spaces between the columns.
272  for (int j = 0; j < dWidth - 1; ++j)
273  {
274  //Zoom in on a point 0 at i*skip, j*skip and its
275  //downsampled neighbor on the right, point 1, its
276  //downsampled neighbor below, point 2, and its
277  //downsampled neighbor below and to the right, point 3.
278  //Like this:
279  // 0 1
280  // 2 3
281  //Indices into the mask.
282  int index0 = width*i*skip + j*skip;
283  int index1 = index0 + skip;
284  int index2 = index0 + width*skip;
285  int index3 = index2 + skip;
286  //Indices into the vertices.
287  int vertID0 = dWidth*i + j;
288  int vertID1 = vertID0 + 1;
289  int vertID2 = vertID0 + dWidth;
290  int vertID3 = vertID2 + 1;
291 
292  //Can I connect point 1 to point 2?
293  if (mask.testBit(index1) & mask.testBit(index2))
294  {
295  // I use the default / method to divide the
296  //triangles.
297  if(mask.testBit(index0)) //Can I connect 0 to 1 and 2?
298  {
299  //If so, I can make a triangle out of them.
300  indices.push_back(vertID0);
301  indices.push_back(vertID1);
302  indices.push_back(vertID2);
303  computeTriangleNormal(normals, xyz, vertID2,
304  vertID0, vertID1);
305  }
306  if(mask.testBit(index3)) //Can I connect 3 to 1 and 2?
307  {
308  indices.push_back(vertID2);
309  indices.push_back(vertID3);
310  indices.push_back(vertID1);
311  computeTriangleNormal(normals, xyz, vertID2,
312  vertID1, vertID3);
313  }
314  }
315  else if (mask.testBit(index0) & mask.testBit(index3)) //Can I connect 0 and 3?
316  {
317  //I use the \ method the divide the triangles.
318  if(mask.testBit(index1)) //Can I connect 1 to 0 and 3?
319  {
320  indices.push_back(vertID1);
321  indices.push_back(vertID3);
322  indices.push_back(vertID0);
323  computeTriangleNormal(normals, xyz, vertID3,
324  vertID0, vertID1);
325  }
326  if(mask.testBit(index2)) //Can I connect 2 to 0 and 3?
327  {
328  indices.push_back(vertID2);
329  indices.push_back(vertID3);
330  indices.push_back(vertID0);
331  computeTriangleNormal(normals, xyz, vertID3,
332  vertID2, vertID0);
333  }
334  }
335  }
336  }
337 
338  //Allocate the ibo.
339  ibo = QGLBuffer(QGLBuffer::IndexBuffer);
340  ibo.create();
341  ibo.bind();
342  ibo.setUsagePattern(QGLBuffer::StaticDraw);
343  ibo.allocate(indices.data(), sizeof(GLuint)*indices.size());
344  nTriangles = indices.size()/3;
345 
346  //Allocate the nbo.
347  nbo.create();
348  nbo.bind();
349  nbo.setUsagePattern(QGLBuffer::StaticDraw);
350  nbo.allocate(normals.data(), sizeof(GLfloat)*normals.size());
351 
352  delete [] xyz;
353 }

355 void
356 RangeImageRenderer::computeTriangleNormal(QVector<GLfloat>&
357  normals, const float* xyz, int centerIdx,
358  int leftIdx, int rightIdx)
359 {
360  //WARNING: This modifies the normals QVector.
361  //Doing this to avoid having to store the normals on
362  //both CPU and GPU.
363  QVector3D centerPt (xyz[3*centerIdx],
364  xyz[3*centerIdx + 1], xyz[3*centerIdx + 2]);
365  QVector3D leftPt (xyz[3*leftIdx],
366  xyz[3*leftIdx + 1], xyz[3*leftIdx + 2]);
367  QVector3D rightPt (xyz[3*rightIdx],
368  xyz[3*rightIdx + 1], xyz[3*rightIdx + 2]);
369  QVector3D normalVec = QVector3D::normal(
370  rightPt - centerPt, leftPt - centerPt);
371 
372 
373  //Add normalVec for the triangle to the normals
374  //for each participating vertex in the triangle.
375  //This will generate the "average" normal vector
376  //of all adjacent faces to each vertex.
377  //It will have the proper direction but need to be normalized.
378  normals[3*centerIdx] += normalVec.x();
379  normals[3*centerIdx + 1] += normalVec.y();
380  normals[3*centerIdx + 2] += normalVec.z();
381  normals[3*leftIdx] += normalVec.x();
382  normals[3*leftIdx + 1] += normalVec.y();
383  normals[3*leftIdx + 2] += normalVec.z();
384  normals[3*rightIdx] += normalVec.x();
385  normals[3*rightIdx + 1] += normalVec.y();
386  normals[3*rightIdx + 2] += normalVec.z();
387 }

389 void
390 RangeImageRenderer::internalRender(GraphicsWidget* scene)
391 {
392  //Init buffers & shaders if this is first draw
393  //Need context to do this.
394  if (!vbo.isCreated())
395  {
396  qDebug() << "Drawing for the first time....";
397  initShaders();
398  //Turn on the vertex array capability.
399  glEnableClientState(GL_VERTEX_ARRAY);
400  //Turn on the normal array capability.
401  glEnableClientState(GL_NORMAL_ARRAY);
402  //Turn on the texture coordinate array capability.
403  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
404  initBuffers();
405  glEnable(GL_BLEND); //for the selection plane.
406  }
407 
408  //Bind current shader.
409  if (shaderPrograms.isEmpty())
410  {
411  //In case you've done something like pull all
412  //the programs out of shaderPrograms:
413  qDebug() << "You have no shader programs!" <<
414  "Drawing with 1.0 pipeline.";
415  glColor3f(0, 0, 1.0f); //blue pen
416  glPolygonMode(GL_FRONT_AND_BACK,
417  GL_FILL);
418  }
419  else
420  {
421  shaderPrograms[currentShaderProgram]->bind();
422  //Provide shaders with any needed data/settings.
423  passDataToShaders(scene);
424  //If currentShaderProgram is valid, then polyMode
425  //will have an entry for it (see setDrawMode).
426  glPolygonMode(GL_FRONT_AND_BACK,
427  polyModes[currentShaderProgram]);
428  }
429 
430  //Set up matrix stack.
431  glMatrixMode(GL_MODELVIEW);
432  glPushMatrix();
433  glMultMatrixd(transform.constData());
434  glMultMatrixd(model->getCoordinateSystemMatrix().constData());
435 
436  //Draw model data.
437  vbo.bind();
438  //Put buffer contents in Vertex Array.
439  //NULL says "use the currently bound buffer"
440  glVertexPointer(3, GL_FLOAT, 0, NULL);
441 
442  nbo.bind();
443  //Put buffer contents in Vertex Array.
444  //NULL says "use the currently bound buffer"
445  glNormalPointer(GL_FLOAT, 0, NULL);
446 
447  tbo.bind();
448  //Put buffer contents in Vertex Array.
449  //NULL says "use the currently bound buffer"
450  glTexCoordPointer(2, GL_FLOAT, 0, NULL);
451 
452  ibo.bind();
453  //Draw the buffers in the order given in ibo.
454  //NULL says "use the currently bound buffer" as the indices.
455  glDrawElements(GL_TRIANGLES, nTriangles*3,
456  GL_UNSIGNED_INT, NULL);
457 
458  //Unbind current shader program.
459  if (!shaderPrograms.isEmpty())
460  shaderPrograms[currentShaderProgram]->release();
461 
462  //Draw selection, if enabled.
463  if (selection->isEnabled())
464  selection->draw();
465 
466  glMatrixMode(GL_MODELVIEW);
467  glPopMatrix();
468 
469  //Draw coordinate system.
470  if (drawCS)
471  {
472  glDisable(GL_DEPTH_TEST);
473  glMatrixMode(GL_MODELVIEW);
474  glPushMatrix();
475  glMultMatrixd(transform.constData());
476  float lineLength = 4e3;
477  glLineWidth(2.0f);
478  glBegin(GL_LINES);
479  glColor3f(1.0f, 0, 0); //red pen
480  glVertex3f(0, 0, 0);
481  glVertex3f(lineLength, 0, 0); //x
482  glColor3f(0, 1.0f, 0); //green pen
483  glVertex3f(0, 0, 0);
484  glVertex3f(0, lineLength, 0); //y
485  glColor3f(0, 0, 1.0f); //blue pen
486  glVertex3f(0, 0, 0);
487  glVertex3f(0, 0, lineLength);
488  glEnd();
489  glEnable(GL_DEPTH_TEST);
490  glMatrixMode(GL_MODELVIEW);
491  glPopMatrix();
492  }
493 }

495 void
496 RangeImageRenderer::draw(GraphicsWidget* scene)
497 {
498  if (needSelectionUpdate)
499  {
500  //Save state.
501  int oldProgram = currentShaderProgram;
502  bool oldDrawCS = drawCS;
503  bool oldSelectionEnabled = selection->isEnabled();
504 
505  //Draw for identifying window coordinates.
506  currentShaderProgram = 4;
507  drawCS = false;
508  selection->setEnabled(false);
509  internalRender(scene);
510 
511  //Read back pixel color.
512  unsigned char color[3];
513  //Must flip y - Qt and OpenGL are different.
514  glReadPixels(winX,
515  scene->height() - winY, 1, 1,
516  GL_RGB, GL_UNSIGNED_BYTE, color);
517 
518  //Determine i and j values from pixel color.
519  int id = (65536*color[0]) + (256*color[1]) + color[2];
520  int i = id/width;
521  int j = id%width;
522 
523  //Update selection.
524  selection->updateSelectionBasis(pixelSizeX*j, pixelSizeY*i);
525 
526  //Restore state.
527  currentShaderProgram = oldProgram;
528  drawCS = oldDrawCS;
529  selection->setEnabled(oldSelectionEnabled);
530  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
531  //We have updated the selection.
532  needSelectionUpdate = false;
533  }
534  internalRender(scene);
535 }
