32 ManipulatePlate::ManipulatePlate(RangeImage* newPlate, float* plateDepth)
33 {
34  //Cache some data.
35  width = newPlate->getWidth();
36  height = newPlate->getHeight();
37  pixelSizeX = newPlate->getPixelSizeX();
38  pixelSizeY = newPlate->getPixelSizeY();
39  depth = newPlate->getDepth();
40  mask = newPlate->getMask();
41 
42  //Fit a plane to the plate data.
43  QVector<int> idx;
44  //Go in a diagonal line across the plate.
45  //Take downsampled points (40 of them).
46  int skip = height/40;
47  for (int i = 0; i < height - skip; i += skip)
48  {
49  //Left diagonal.
50  int j = (int) i * ((float) width/(float) height);
51  int id = i*width + j;
52  if (mask.testBit(id))
53  {
54  idx.push_back(id);
55  }
56 
57  //Right diagonal.
58  j = width - j;
59  id = i*width + j;
60  if (mask.testBit(id))
61  {
62  idx.push_back(id);
63  }
64  }
65  QVector4D planeCoeffs = fitPlane(idx);
66  float nx = planeCoeffs.x();
67  float ny = planeCoeffs.y();
68  float nz = planeCoeffs.z();
69  float p = planeCoeffs.w();
70 
71  //Subtract off the plane.
72  for (int i = 0; i < height; ++i)
73  {
74  for (int j = 0; j < width; ++j)
75  {
76  float x = j*pixelSizeX;
77  float y = i*pixelSizeY;
78  plateDepth[i*width + j]
79  -= (-1/nz)*(p + (nx*x) + (ny*y));
80  }
81  }
82 }

90 ManipulatePlate::fitPlane(const QVector<int>& idx)
91 {
92  //Here we solve the least squares problem
93  //to determine a, b, & c in the equation
94  //z= ax + by + c.
95  //We can format this as so:
96  //+ + + + + +
97  //|x0 y0 1| |a| = |z0|
98  //|x1 y1 1| |b| |z1|
99  //|x2 y2 1| |c| |z2|
100  //|x3 y3 1| + + |z3|
101  //|. . .| | .|
102  //|. . .| | .|
103  //|. . .| | .|
104  //|xn yn 1| |zn|
105  //+ + + +
106  //Which we can write as Xv = z.
107  //Then we can take (X^T)Xv = (X^T)z.
108  //Then we solve v = ((X^T)X)^(-1)*(X^T)z.
109  //Reference: http://tutorial.math.lamar.edu/Classes/LinAlg/LeastSquares.aspx
110 
111  QVector<float> X, z;
112  for (int i = 0; i < idx.size(); ++i)
113  {
114  //Load a row xi yi 1 into X
115  X.push_back((idx[i]%width)*pixelSizeX); //x
116  X.push_back((idx[i]/width)*pixelSizeY); //y
117  X.push_back(1.0f);
118 
119  //Load a depth value into z vector.
120  z.push_back(depth[idx[i]]);
121  }
122 
123  //(X^T)X
124  QMatrix4x4 XTX; //Identity matrix.
125  for (int i = 0; i < 3; ++i)
126  {
127  for (int j = 0; j < 3; ++j)
128  {
129  XTX(i, j) = 0;
130  for (int k = 0; k < idx.size(); ++k)
131  {
132  //This makes sense if you look at
133  //doing (X^T)X by hand.
134  XTX(i, j) += X[k*3 + i]*X[k*3 + j];
135  }
136  }
137  }
138 
139  //(X^T)z
140  QVector4D XTz; //zero vector.
141  for (int k = 0; k < idx.size(); ++k)
142  {
143  XTz += QVector4D(X[k*3]*z[k], //xk*zk
144  X[k*3 + 1]*z[k], //yk*zk
145  z[k], //zk
146  0); //w, has no effect
147  }
148 
149  //Solve for v
150  //A note on ((X^T)X)^(-1):
151  //Using the properties of block matrices, it is simple to
152  //show that
153  //+ ++ +
154  //|A 0||A^(-1) 0| is the identity matrix,
155  //|0^T 1||0^T 1|
156  //+ ++ +
157  //where the 0's are the appropriate column vectors of 0's.
158  //Therefore, we can use the QMatrix4x4 object to do
159  //the inversion and multiplication.
160  //Reference: http://mathworld.wolfram.com/BlockMatrix.html
161  //The QVector3D constructor drops the meaningless w coordinate.
162  QVector3D v (XTX.inverted()*XTz);
163 
164  //v = [a, b, c]^T; z = ax + by + c
165  //Put into plane form to find normal:
166  //-ax - by + 1z - c = 0
167  //Therefore normal is [-a -b 1]^T normalized.
168  QVector3D n = -v;
169  float negC = n.z();
170  n.setZ(1);
171  negC /= n.length();
172  n.normalize();
173 
174  return QVector4D(n.x(), n.y(), n.z(), negC);
175 }
