35 ComputeFlatScrewdriverTipCsys::ComputeFlatScrewdriverTipCsys(
36  int w, int h, float pixX, float pixY,
37  const QVector<float>& zData, const QBitArray& maskData,
38  float yScanAngle, QWidget *parent):
39  QObject(parent)
40 {
41  //Do some caching.
42  mask = maskData;
43  depth = zData;
44  width = w;
45  height = h;
46  pixelSizeX = pixX;
47  pixelSizeY = pixY;
48 
49  //Make an estimate of the tip edge.
50  findEdgeIndices();
51 
52  //Store a translation to the centroid of the tip edge estimate
53  for (int i = 0; i < edgeIdx.size(); ++i)
54  {
55  float xi = (edgeIdx[i]%width)*pixelSizeX;
56  float yi = (edgeIdx[i]/width)*pixelSizeY;
57  centroid += QVector3D(xi, yi, depth[edgeIdx[i]]);
58  }
59  centroid /= edgeIdx.size();
60  translationMatrix.translate(-centroid);
61 
62  //Estimate the axes
63  //y axis.
64  generateYAxis();
65  yAxisVector = -yAxisVector;
66  //x axis.
67  QMatrix4x4 rotateXAxis;
68  rotateXAxis.rotate(yScanAngle, QVector3D(0, 1, 0));
69  xAxisVector = rotateXAxis * QVector3D(0, 0, 1);
70  xAxisVector = -xAxisVector;
71  //z axis.
72  zAxisVector = QVector3D::crossProduct(
73  xAxisVector, yAxisVector);
74  zAxisVector.normalize();
75 
76  // The following operation is to ensure that
77  // the new coordinate system is orthonormal
78  xAxisVector = QVector3D::crossProduct(
79  yAxisVector, zAxisVector);
80  xAxisVector.normalize();
81 
82  //Construct final coordinate system.
83  QMatrix4x4 basisChange; //identity matrix
84  //QVector4D constructor sets w to 0
85  basisChange.setColumn(0, QVector4D(xAxisVector));
86  basisChange.setColumn(1, QVector4D(yAxisVector));
87  basisChange.setColumn(2, QVector4D(zAxisVector));
88  coordinateSystem = basisChange.inverted() * translationMatrix;
89 }

96 // find the edge point horizontally
97 // assume edge is the largest depth value and find roughly 300 points
98 // on that edge (NOTE: a number makes sense)
99 void
100 ComputeFlatScrewdriverTipCsys::findEdgeIndices()
101 {
102  edgeIdx.clear(); //Get rid of any lingering data.
103 
104  //Amount to downsample in the row direction.
105  //If 1, take every row; if 2, take every other row, etc.
106  int skip = height/300 + 1;
107 
108  //Find maxes (unless they are too close to the edge).
109  for (int i = 0; i < height; i += skip)
110  {
111  //Go down row & find max point.
112  int lineId = i * width;
113  float tmp = -FLT_MAX;
114  int maxIdx = 0;
115  for (int j = 0; j < width; j++)
116  {
117  int idx = lineId + j;
118  if (mask.testBit(idx))
119  {
120  if (tmp < depth[idx])
121  {
122  maxIdx = idx;
123  tmp = depth[idx];
124  }
125  }
126  }
127  //If the maxIdx is valid and does not occur near the edge,
128  //store it.
129  if ( maxIdx > 0 &&
130  ((maxIdx % width) > 10) &&
131  ((maxIdx % width) < width - 10)
132  )
133  {
134  edgeIdx.push_back(maxIdx);
135  }
136  }
137 }

139 // define y direction by 3D line fitting through linear regression
140 // use the method described in Wiki: http://en.wikipedia.org/wiki/User:Vossman/3D_Line_Regression
141 void
142 ComputeFlatScrewdriverTipCsys::generateYAxis()
143 {
144  QMatrix4x4 A; //identity matrix.
145  A.fill(0); //Set to all zeros.
146  QVector4D B; //zero vector.
147  int edgeLength = edgeIdx.size();
148 
149  for (int i = 0; i < edgeLength; ++i)
150  {
151  // swap (x0 y0 z0) to (y z x) to make sure that the line
152  // direction (close to y0 direction) pass through x-y plane,
153  // change y to z.
154  float y = (edgeIdx[i]%width)*pixelSizeX - centroid.x();//x
155  float z = (edgeIdx[i]/width)*pixelSizeY - centroid.y();//y
156  float x = depth[edgeIdx[i]] - centroid.z(); //z
157 
158  A(0,0) += z*z / edgeLength;
159  A(0,2) += z / edgeLength;
160  B += QVector4D(x*z / edgeLength,
161  x / edgeLength,
162  y*z / edgeLength,
163  y / edgeLength);
164  }
165 
166  A(1, 0) = A(0, 2);
167  A(2, 1) = A(0, 0);
168  A(2, 3) = A(0, 2);
169  A(3, 1) = A(0, 2);
170  A(1, 2) = 1;
171  A(3, 3) = 1;
172 
173  QVector4D v0;
174  v0 = A.inverted()*B;
175 
176  // deterine y vector obtain the vector, and
177  // swap back to original coordinate system
178  yAxisVector = QVector3D(v0.y()-v0.w(), 1.0f, v0.x()-v0.z());
179  yAxisVector.normalize();
180 }
