49 void CFindValidMarkRange::LocateMarkRegion(unsigned char *image, unsigned char *qualityData, unsigned char *mask, float &markAngle)
50 {
51  m_maskData = mask;
52 
53  CFilterImage filter(m_imageWidth, m_imageHeight);
54  float *nimage = new float [m_imageSize];
55 
56  memset(mask, 255, sizeof(mask[0])*m_imageSize);
57  NormalizeImageU8(image, nimage, mask);
58  ComputeMask(nimage, qualityData, 0.05f, 200);
59 
60  float *Fx = new float [m_imageSize];
61  float *Fy = new float [m_imageSize];
62  float mFx = ComputeGradientX(nimage, Fx);
63  float mFy = ComputeGradientY(nimage, Fy);
64  markAngle = atan2(-mFy, mFx);
65 
66  // Get absolute value of the gradient map
67  AbsoluteImage(Fx, Fx);
68  AbsoluteImage(Fy, Fy);
69 
70  // binarize the image
71  BinarizeImageFT(Fx, Fx, mask, 0.04f); // 0.04 originally (try 2 if things don't work out)
72  BinarizeImageFT(Fy, Fy, mask, 0.04f); // 0.04 originally
73 
74  // apply median filter
75  filter.MedianFilterFloat(Fx, Fx, 3, mask);
76  filter.MedianFilterFloat(Fy, Fy, 3, mask);
77 
78  // compute absolute difference
79  float * diff = new float [m_imageSize];
80  for (int i = 0; i < m_imageSize; i++)
81  {
82  diff[i] = fabs(Fx[i] - Fy[i]);
83  }
84 
85  // binarize image
86  BinarizeImageFT(diff, diff, mask, 0.15f);
87 
88  // median filtering it again
89  filter.MedianFilterFloat(diff, diff, 3, mask);
90  filter.MedianFilterFloat(diff, diff, 3, mask);
91 
92  // Remove outside boundary by making mask value to be zeros;
93  RemoveBoundary(diff, mask);
94 
95  delete [] diff;
96  delete [] Fx;
97  delete [] Fy;
98 }

127 void CFindValidMarkRange::RemoveBoundary(float *diff, unsigned char *mask)
128 {
129  // Count how many valid pixels per row
130  vector<int>vcont;
131  vcont.clear();
132  for (int i = 0; i < m_imageHeight; i++)
133  {
134  float cnt = 0;
135  for (int j = 0; j < m_imageWidth; j++)
136  {
137  cnt += diff[i*m_imageWidth + j];
138  }
139  //printf("%.2f\t", cnt);
140  int t = cnt > m_imageWidth /2 ? 1 : 0; //less than half valid, the line is not stripe
141  vcont.push_back(t);
142  }
143 
144  int startID = 10;
145  int endID = m_imageHeight - 10;
146  // Find the top boundary
147  for (size_t i = 10; i < vcont.size() - 10; i++)
148  {
149  startID = i;
150  if (vcont[i]) break;
151 
152  }
153 
154  for (size_t i = vcont.size() - 10; i > 10; i--)
155  {
156  endID = i;
157  if (vcont[i]) break;
158  }
159  vcont.clear();
160 
161  //printf("start = %d\t end = %d\n", startID, endID);
162  memset(mask, 0, sizeof(mask[0])* startID * m_imageWidth);
163  memset(&mask[endID * m_imageWidth], 0, sizeof(mask[0])* (m_imageHeight - endID) * m_imageWidth);
164 }

167 void CFindValidMarkRange::NormalizeImage(float *input, float *output, unsigned char *mask)
168 {
169  bool isMask = true;
170  if (mask == NULL)
171  {
172  isMask = false;
173  mask = new unsigned char [m_imageSize];
174  memset(mask, 255, sizeof(mask[0]) * m_imageSize);
175  }
176 
177  float maxV = -FLT_MAX;
178  float minV = FLT_MAX;
179  for(int i = 0; i < m_imageSize; i++)
180  {
181  if (mask[i])
182  {
183  maxV = maxV > input[i] ? maxV : input[i];
184  minV = minV < input[i] ? minV : input[i];
185  }
186  }
187  float irange = 1.0f/(maxV - minV);
188  for(int i = 0; i < m_imageSize; i++)
189  {
190  if (mask[i])
191  {
192  output[i] = ((input[i] - minV)*irange);
193  }
194  else output[i] = 0;
195  }
196 
197  if (!isMask)
198  {
199  delete [] mask;
200  mask = NULL;
201  }
202 }

278 float CFindValidMarkRange::ComputeGradientX(float *image, float *Fx)
279 {
280  float mFx = 0.0f;

282  memset(Fx, 0, sizeof(Fx[0]) * m_imageSize);
283 
284  for (int j = 0; j < m_imageHeight; j++)
285  {
286  int lineId = j * m_imageWidth;
287  for (int i = 1; i < m_imageWidth; i++)
288  {
289  int idx = lineId + i;
290  if (m_maskData[idx] && image[idx-1])
291  {
292  float t = image[idx] - image[idx-1];
293  Fx[idx] = t;
294  mFx += t;
295  cnt++;
296  }
297  }
298  }
299  if (!cnt) return 0.0f;
300  return mFx/cnt;
