74 graphicsWidget::paintEvent(QPaintEvent* event)
75 {
76  QPainter painter (this);
77  QRect drawRect = event->rect();
78 
79  //Draw background.
80  QLinearGradient background (drawRect.topLeft(), drawRect.bottomRight());
81  background.setColorAt(0.0, Qt::darkGray);
82  background.setColorAt(1.0, Qt::black);
83  painter.fillRect(drawRect, background);
84 
85  //Draw texture and mask (if loaded)
86  if ((!texture.isNull()) && (!maskImage.isNull()))
87  {
88  QImage scaledTexture;
89  QImage scaledMask;
90 
91  float winAspect = ((float) width())/((float) height());
92  float textAspect= ((float) textureWidth)/((float) textureHeight);
93  if (textAspect > winAspect) //texture is wider than window.
94  {
95  scaledTexture = texture.scaledToWidth(width());
96  scaledMask = maskImage.scaledToWidth(width());
97  toWidth = true;
98  }
99  else
100  {
101  scaledTexture = texture.scaledToHeight(height());
102  scaledMask = maskImage.scaledToHeight(height());
103  toWidth = false;
104  }
105 
106  //Draw texture and then maskImage.
107  painter.drawImage(drawRect, scaledTexture, drawRect);
108  painter.drawImage(drawRect, scaledMask, drawRect);
109  }

112 void
113 graphicsWidget::drawTo(const QPoint endpoint)
114 {
115  if (maskImage.isNull()) return; //Do not draw if no image.
116 
117  //Calculate the scale of the image for coordinate transformation.
118  float scale;
119  if (toWidth)
120  {
121  scale = ((float) textureWidth)/((float) width());
122  }
123  else
124  {
125  scale = ((float) textureHeight)/((float) height());
126  }
127 
128  //Open the painter on the maskImage.
129  QPainter painter (&maskImage);
130  //Allows us to paint without overlapping and to erase.
131  painter.setCompositionMode(QPainter::CompositionMode_Source);
132  //Setup the pen.
133  QPen toolPen;
134  if (drawMode)
135  {
136  toolPen.setColor(maskColor);
137  }
138  else
139  {
140  toolPen.setColor(emptyColor);
141  }
142  toolPen.setWidth(toolSize*scale);
143  toolPen.setCapStyle(Qt::RoundCap);
144  toolPen.setJoinStyle(Qt::RoundJoin);
145  painter.setPen(toolPen);
146 
147  //Draw the line.
148  painter.drawLine(capturedPos*scale, endpoint*scale);
149  modified = true;
150 
151  //Update the view.
152  update();
153 }

191 void
192 graphicsWidget::setFile(QString filename)
193 {
194  QBitArray mask; //to hold the mask temporarily.
195 
196  //Load file.
197  if (filename.endsWith("mt",
198  Qt::CaseInsensitive)) //file ext is MT or mt
199  {
200  //Delete any already loaded files.
201  delete fileContents;
202 
203  //Load new file
204  fileContents = RangeImage::import(filename);
205  if (NULL == fileContents)
206  {
207  QMessageBox::critical(this, tr("Error"),
208  tr("MT File failed to load."));
209  return;
210  }
211  texture = fileContents->getTexture();
212  mask = fileContents->getMask();
213  }
214  else
215  {
216  QMessageBox::warning(this, tr("Error"),
217  tr("Invalid file format.\n"
218  "Format must be .mt."));
219  return;
220  }
221 
222  //Load texture width and height.
223  textureWidth = texture.width();
224  textureHeight = texture.height();
225 
226  //Encode mask to QImage.
227  //This will render the masked portion as transparent coral for now.
228  maskImage = QImage (textureWidth, textureHeight,
229  QImage::Format_ARGB32); //_Premultiplied);
230  for (int i = 0; i < textureHeight; ++i)
231  {
232  for (int j = 0; j < textureWidth; ++j)
233  {
234  //Masked region is 0
235  //Kept (non-masked) region is 255
236  if (!mask.testBit((textureWidth*i) + j)) //If mask is 0.
237  {
238  maskImage.setPixel(j, i, maskColor.rgba());
239  }
240  else
241  {
242  maskImage.setPixel(j, i, emptyColor.rgba());
243  }
244  }
245  }
246 
247  //Rotate the images for easier viewing.
248  QMatrix rot;
249  rot.rotate(90);
250  maskImage = maskImage.transformed(rot);
251  texture = texture.transformed(rot);
252  int temp = textureWidth;
253  textureWidth = textureHeight;
254  textureHeight = temp;
255 
256  update(); //Refresh widget.
257 }

259 bool
260 graphicsWidget::saveFile(QString filename)
261 {
262  if (NULL == fileContents) return false;
263 
264  //Decode mask from maskImage.
265  //Undo the rotation.
266  QMatrix rot;
267  rot.rotate(-90);
268  maskImage = maskImage.transformed(rot);
269  int temp = textureWidth;
270  textureWidth = textureHeight;
271  textureHeight = temp;
272  //Create mask, initialized to false.
273  QBitArray mask (textureWidth*textureHeight, false);
274  for (int i = 0; i < textureHeight; ++i)
275  {
276  for (int j = 0; j < textureWidth; ++j)
277  {
278  if (maskImage.pixel(j, i) == emptyColor.rgba())
279  {
280  mask.setBit((textureWidth*i) + j);
281  }
282  }
283  }
284  //Redo the rotation.
285  rot = QMatrix();
286  rot.rotate(90);
287  maskImage = maskImage.transformed(rot);
288  temp = textureWidth;
289  textureWidth = textureHeight;
290  textureHeight = temp;
291 
292  //Create new range image and delete old one.
293  //Make mutable copies of the depth, texture, csys.
294  QVector<float> newDepth (fileContents->getDepth());
295  QImage newTexture (fileContents->getTexture());
296  QMatrix4x4 newCsys (fileContents->getCoordinateSystemMatrix());
297 
298  //Make the new data structure.
299  RangeImage* newContents = new RangeImage(
300  fileContents->getWidth(),
301  fileContents->getHeight(),
302  fileContents->getPixelSizeX(),
303  fileContents->getPixelSizeY(),
304  newDepth,
305  newTexture,
306  mask,
307  newCsys);
308  delete fileContents;
309  fileContents = newContents;
310 
311  //Run connected components to avoid floating data islands.
312  Clean cleaner;
313  newContents = cleaner.connectedComponents(fileContents);
314  delete fileContents;
315  fileContents = newContents;
316 
317  //Perform any desired cleaning functions prior to save.
318  cleaningDialog options;
319  options.exec();
320  if (options.getSpike())
321  {
322  qDebug() << "Removing spikes and filling holes.";
323  //This redoes the conncomps if necessary.
324  newContents = cleaner.spikeRemovalHoleFilling(
325  fileContents);
326  delete fileContents;
327  fileContents = newContents;
328  }
329  if (options.getDetrend())
330  {
331  qDebug() << "Detrending....";
332  newContents = cleaner.detrend(fileContents);
333  delete fileContents;
334  fileContents = newContents;
335  }
336  if (options.getComputeCsys())
337  {
338  if (options.getCentroid())
339  {
340  qDebug() << "Computing a centroid coordinate system.";
341  newContents = cleaner.coordinateSystem2Centroid(
342  fileContents);
343  delete fileContents;
344  fileContents = newContents;
345  }
346  else
347  {
348  qDebug() << "Computing a tip coordinate system.";
349  newContents = cleaner.coordinateSystem4Tip(
350  fileContents);
351  delete fileContents;
352  fileContents = newContents;
353  }
354  }
355 
356  //Save
357  int success = fileContents->save(filename);
358  if (!success) return false;
359 
360  //If we got here, the save was successful.
361  modified = false;
362  return true;
363 }
