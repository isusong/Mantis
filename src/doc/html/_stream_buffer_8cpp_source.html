71 void
72 StreamBuffer::flush()
73 {
74  //qDebug() << "flushing stream buffer.";
75  //qDebug() << "vertices pointer: " << vertices;
76  //Number of tries for mapping and unmapping before giving up.
77  int maxTries = 10; //Beyond this doesn't make sense.
78 
79  //Get the buffer pointer from the GPU.
80  GLfloat* bufferMap;
81  for (int i = 0; i < maxTries; ++i)
82  {
83  bufferMap = (GLfloat*) vbo.map(QGLBuffer::WriteOnly);
84  if (bufferMap != NULL) break;
85  }
86 
87  if (bufferMap == NULL)
88  {
89  qDebug() << "Buffer couldn't be mapped; " <<
90  "mark will be incomplete.";
91  }
92  else
93  {
94  bool unmapSuccess;
95  for (int i = 0; i < maxTries; ++i)
96  {
97  std::memcpy(bufferMap, vertices, 4*size*sizeof(GLfloat));
98  unmapSuccess = vbo.unmap();
99  if (unmapSuccess) break;
100  }
101 
102  if (unmapSuccess)
103  {
104  //Draw the buffer.
105  //Put buffer contents in Vertex Array.
106  //NULL says "use the currently bound buffer"
107  //qDebug() << "Putting buffer contents in vertex array.";
108  glEnableClientState(GL_VERTEX_ARRAY);
109  glVertexPointer(4, GL_FLOAT, 0, NULL);
110  //qDebug() << "Done putting.";
111  //qDebug() << "size = " << size;
112  //qDebug() << "capacity = " << capacity;
113  glDrawArrays(drawMode, 0, size);
114  //qDebug() << "Drew arrays.";
115  }
116  else
117  {
118  qDebug() << "Buffer couldn't be unmapped;" <<
119  "mark will be incomplete.";
120  }
121  }
122 
123  //Start over-writing vertices.
124  size = 0;
125  //qDebug() << "Done flushing stream buffer.";
126 }
