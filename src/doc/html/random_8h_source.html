140 void Select(int n, int N, int* result) {
141  // Algorithm S, Knuth v. 2, p. 137
142  // select random combination
143 
144  // select n items at random from a set of N, 0 < n <= N
145  // each item is selected (in order) with prob n/N
146  // items are numbered [0, ..., N-1]
147  // returns result in array int result[0,...,n-1]
148  // (caller must allocate space for array)
149 
150  // initialize
151  int t = 0; // number of items dealt with
152  int m = 0; // number of items selected
153  float U;
154 
155  while (m < n) {
156  U = rnd(0,1); // uniformly generated random number
157  // on interval (0,1)
158 
159  if ( ((N - t) * U) >= (n - m) ) {
160  // skip this item
161  t++;
162  }
163  else {
164  // select next item and add it to list
165  result[m++] = t++;
166  }
167  }
168 }
