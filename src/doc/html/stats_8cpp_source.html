52 {

134 pair<double, double>
135 computeRankSumSqSum(const vector<T>& values1, const vector<T>& values2)
136 {
137  vector<Ordinal<T> > ordinals;
138  ordinals.reserve(values1.size() + values2.size());
139  appendAsOrdinalsTo(values1, ordinals, FIRST);
140  appendAsOrdinalsTo(values2, ordinals, SECOND);
141 
142  sort(ordinals.begin(), ordinals.end());
143 
144  assert(ordinals.size() == values1.size() + values2.size());
145 
146  double rankSum1 = 0.0;
147  double sqRankSum = 0.0;
148 
149  int size = ordinals.size();
150  int numberEqual[2];
151 
152  for (int i = 0; i < size;) {
153  Ordinal<T>& o = ordinals[i];
154  int tag = o.tag();
155 
156  assert(tag == 1 || tag == 0);
157 
158  numberEqual[tag] = 1;
159  numberEqual[1 - tag] = 0;
160  for (int j = i + 1; j < size && o == ordinals[j]; ++j) {
161  int tag = ordinals[j].tag();
162  ++numberEqual[tag];
163  }
164 
165  int totalNumberEqual = numberEqual[0] + numberEqual[1];
166 
167  //rank r = i + 1
168  //sumOfRanks = r + (r+1) + ... + (r+totalNumberEqual-1)
169  // = totalNumberEqual * (r + (r+totalNumberEqual-1))/2
170  // = totalNumberEqual * ((i+1) +((i+1)+totalNumberEqual-1))/2
171  // = totalNumberEqual * (2*i + 1 + totalNumberEqual)/2
172 
173  double sumOfRanks = (totalNumberEqual * (2*i + 1 + totalNumberEqual))/2;
174 
175  double averageRank = sumOfRanks/totalNumberEqual;
176 
177  rankSum1 += averageRank * numberEqual[0];
178  sqRankSum += totalNumberEqual * averageRank * averageRank;
179  i += totalNumberEqual;
180  }
181  return make_pair(rankSum1, sqRankSum);
182 }

188 double t1Statistic(const vector<double>& x, const vector<double>& y)
189 {
190  double n = x.size();
191  double m = y.size();
192  double N = m + n;
193  pair<double, double> r(computeRankSumSqSum(x,y));
194  double T = r.first;
195  double R2 = r.second;
196  double N1 = N + 1;
197  double top = T - n * N1/2.0;
198  double bot2 = n*m/(N*(N-1))*R2 - n*m*N1*N1/(4*(N-1));
199  return top / sqrt(bot2);
