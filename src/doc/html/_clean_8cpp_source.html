62 //Threshold based on the quality map and the texture.
63 //Does not delete tip pointer.
64 QBitArray
65 Clean::threshold(RangeImage* tip, const QImage& qualityMap,
66  int quality_threshold, int texture_threshold)
67 {
68  //Cache some items from the tip.
69  int width = tip->getWidth();
70  int height = tip->getHeight();
71  QBitArray mask (tip->getMask()); //Make a copy of the mask.
72  QImage texture = tip->getTexture();
73  int qPixel, tPixel;
74 
75  //Modify the mask copy.
76  for (int i = 0; i < height; ++i)
77  {
78  for (int j = 0; j < width; ++j)
79  {
80  if (mask.testBit(width*i + j)) //If the pixel is on:
81  {
82  //Grab the grayscale value of the pixel quality.
83  //TODO: qGray solution
84  qPixel = qGray(qualityMap.pixel(j, i));
85 
86  //Grab the grayscale value of the pixel texture.
87  //TODO: Dr. Zhang's solution
88  tPixel = maxGray(texture.pixel(j, i));
89 
90  //If the pixel doesn't meet the threshold,
91  //turn it off.
92  if ((tPixel < texture_threshold) ||
93  (qPixel > quality_threshold))
94  mask.clearBit(width*i + j);
95  }
96  }
97  }
98 
99  return mask;
100 }

102 //Finds the border of the unmasked regions and chips away at it.
103 /*
104  * @param lx Number of pixels to be chipped away on left boundary.
105  * @param rx Number of pixels to be chipped away on right boundary.
106  *
107  * @param ty Number of pixels to be chipped away on top boundary.
108  * @param by Number of pixels to be chipped away on bottom boundary.
109  */
110 void
111 Clean::removeAdditionalBoundary(unsigned char* maskData,
112  int imageWidth, int imageHeight, int lx, int rx, int ty, int by)
113 {
114  // left and right
115  for (int i = 0; i < imageHeight; i++)
116  {
117  for (int j = 0; j < imageWidth - lx; j++)
118  {
119  int id = i * imageWidth + j;
120  if (maskData[id])
121  {
122 
123  memset(&maskData[id], 0, sizeof(maskData[0])*lx);
124  j = imageWidth+1;
125  }
126  }
127 
128  for (int j = imageWidth-1; j > rx; j--)
129  {
130  int id = i * imageWidth + j;
131  if (maskData[id])
132  {
133 
134  memset(&maskData[id-rx+1], 0, sizeof(maskData[0])*rx);
135  j = -1;
136  }
137  }
138  }
139 
140  // top and bottom
141  for (int i = 0; i < imageWidth; i++)
142  {
143  for (int j = 0; j < imageHeight - ty; j++)
144  {
145  int id = j * imageWidth + i;
146  if (maskData[id])
147  {
148  for (int k = 0; k < ty; k++)
149  {
150  maskData[(j+k)*imageWidth + i] = 0;
151  }
152  j = imageHeight;
153  }
154  }
155 
156  for (int j = imageHeight-1; j > by; j--)
157  {
158  int id = j * imageWidth + i;
159  if (maskData[id])
160  {
161  for (int k = 0; k < by; k++)
162  {
163  maskData[(j-by+k+1)*imageWidth + i] = 0;
164  }
165  j = by;
166  }
167  }
168  }
169 }

173 RangeImage*
174 Clean::cleanFlatScrewdriverTip(RangeImage* tip,
175  const QImage& qualityMap, int quality_threshold,
176  int texture_threshold)
177 {
178  //Threshold the tip's mask with texture and quality.
179  QBitArray mask = threshold(tip, qualityMap,
180  quality_threshold, texture_threshold);
181 
182  //Cache some data.
183  int width = tip->getWidth();
184  int height = tip->getHeight();
185  float pixelSizeX = tip->getPixelSizeX();
186  float pixelSizeY = tip->getPixelSizeY();
187  QImage texture = tip->getTexture();
188  //Copy the depth into a mutable vector.
189  QVector<float> depth = QVector<float>(tip->getDepth());
190 
191  //Copy the tip mask into an unsigned char array with 255 as "on."
192  QVector<unsigned char> ucharMask;
193  for (int i = 0; i < mask.size(); ++i)
194  ucharMask.push_back(255*mask.testBit(i));
195 
196  //Use connected components to eliminate all but
197  //the largest unmasked area.
198  CFastConnectComponent conComp(width, height);
199  QVector<unsigned char> ucharMaskCopy (ucharMask);
200  conComp.FastConnectAlg(ucharMask.data(),
201  ucharMaskCopy.data(), 0, 255);
202  ucharMask = ucharMaskCopy;
203 
204  //Laura - Not sure about this.
205  //If zData is large spike in positive or
206  //negative direction, set mask and zData there to 0.
207  //for (int i = 0; i < mask.size(); ++i)
208  //{
209  // if (fabs(depth[i]) > 1E9)
210  // {
211  // depth[i] = 0;
212  // ucharMask[i] = 0;
213  // }
214  //}
215 
217  //For now, this doesn't modify anything but
218  //the mask you pass in.
219  QVector<float> xData, yData;
220  for (int j = 0; j < width; ++j)
221  xData.push_back(j*pixelSizeX);
222  for (int i = 0; i < height; ++i)
223  yData.push_back(i * pixelSizeY);
224  CSpikeRemoval spikeRemover;
225  //If the last arg is false, this may modify the depth.
226  spikeRemover.PolyLineRemoval(xData.data(), yData.data(),
227  depth.data(), ucharMask.data(), 100, 100, 7,
228  width, height, true);
229 
231  ucharMaskCopy = ucharMask;
232  conComp.FastConnectAlg(ucharMask.data(),
233  ucharMaskCopy.data(), 0, 255);
234  ucharMask = ucharMaskCopy;
235 
236  //Fill holes in depth data.
237  //Only fills holes with width less than or equal
238  //to the last argument to FillHolesFloat.
239  //FYI: Modifies depth, of course.
240  CFilterImage filter(width, height);
241  filter.FillHolesFloat(depth.data(), ucharMask.data(), 20);
242 
243  // eat in additional few points from outside
244  removeAdditionalBoundary(ucharMask.data(),
245  width, height, 20, 20, 20, 20);
246 
247  //Convert results to QBitArray and make new tip.
248  //If the bit is on in the mask and off in ucharMask,
249  //turn it off in the mask.
250  //Moreover, if the bit is off in the mask and on in
251  //ucharMask, it means that a hole has been filled.
252  //Turn it on in the mask, too.
253  for(int i = 0; i < ucharMask.size(); ++i)
254  {
255  if ((mask.testBit(i)) && (0 == ucharMask[i]))
256  {
257  mask.clearBit(i);
258  }
259  if ((!mask.testBit(i)) && (255 == ucharMask[i]))
260  {
261  mask.setBit(i);
262  }
263  }
264 
265  //Compute coordinate system matrix.
266  //Assume scan at 45 degrees for now.
267  //May change in the future.
268  ComputeFlatScrewdriverTipCsys computeCsys (width, height,
269  pixelSizeX, pixelSizeY, depth, mask);
270  QMatrix4x4 csys = computeCsys.getCoordinateSystemMatrix();
271 
272  return new RangeImage(width, height, pixelSizeX, pixelSizeY,
273  depth, texture, mask, csys);
274 }

288 RangeImage*
289 Clean::cleanStriatedLeadMark(RangeImage* plate, const QImage& qualityMap)
290 {
291  //Cache some data.
292  int width = plate->getWidth();
293  int height = plate->getHeight();
294  float pixelSizeX = plate->getPixelSizeX();
295  float pixelSizeY = plate->getPixelSizeY();
296  QImage texture = plate->getTexture();
297  QBitArray mask (plate->getMask()); //Make a copy of the mask.
298  //Make a copy of the coordinate system.
299  QMatrix4x4 coordinateSystem (plate->getCoordinateSystemMatrix());
300  //Copy the depth into a mutable vector.
301  QVector<float> depth = QVector<float>(plate->getDepth());
302 
303  //Make a grayscale texture copy.
304  QVector<unsigned char> grayTexture;
305  for (int i = 0; i < height; ++i)
306  {
307  for (int j = 0; j < width; ++j)
308  {
309  //Should be row major, according to PngFileIO.cpp.
310  //Int output from maxGray should be between 0 and 255.
311  int grayValue = maxGray(texture.pixel(j, i));
312  if ((255 < grayValue) || (0 > grayValue))
313  {
314  qDebug() << "Plate texture was not in range" <<
315  " [0, 255].";
316  }
317  grayTexture.push_back((unsigned char) grayValue);
318  }
319  }
320 
321  //Copy the mask into an unsigned char array with 255 as "on."
322  QVector<unsigned char> ucharMask;
323  for (int i = 0; i < mask.size(); ++i)
324  {
325  ucharMask.push_back(255*mask.testBit(i));
326  }
327 
328  //Make a grayscale quality map copy.
329  QVector<unsigned char> qualityData;
330  for (int i = 0; i < height; ++i)
331  {
332  for (int j = 0; j < width; ++j)
333  {
334  //Should be row major, according to PngFileIO.cpp.
335  //Int output from maxGray should be between 0 and 255.
336  int grayValue = maxGray(qualityMap.pixel(j, i));
337  if ((255 < grayValue) || (0 > grayValue))
338  {
339  qDebug() << "Plate quality map was not in range" <<
340  " [0, 255].";
341  }
342  qualityData.push_back((unsigned char) grayValue);
343  }
344  }
345 
346  //Find the marked area, mask off the rest.
347  CFindValidMarkRange markRange(width, height);
348  float markAngle = 0;
349  markRange.LocateMarkRegion(grayTexture.data(),
350  qualityData.data(), ucharMask.data(), markAngle);
351 
352  //Median filter the data.
353  CFilterImage filter(width, height);
354  //Make a copy of the mask.
355  QVector<unsigned char> ucharMaskCopy (ucharMask);
356  filter.MedianFilterU8(ucharMask.data(),
357  ucharMaskCopy.data(), 5);
358  //Since we median filtered the mask, we need
359  //to make sure that no bad points became classified
360  //as good points by mistake.
361  for (int i = 0; i < width*height; ++i)
362  {
363  if(255 != ucharMask[i])
364  ucharMaskCopy[i] = 0;
365  }
366 
367  //Find the connected components.
368  qDebug() << "Finding connected components....";
369  CFastConnectComponent conComp(width, height);
370  conComp.FastConnectAlg(ucharMaskCopy.data(),
371  ucharMask.data(), 0, 255);
372 
373  //Remove spikes. WARNING: This modifies depth.
374  qDebug() << "Median filtering the z data....";
375  filter.MedianFilterFloat(depth.data(), depth.data(),
376  5, ucharMask.data());
377 
378  //Fill holes. WARNING: This modifies depth.
379  qDebug() << "Filling holes....";
380  filter.FillHolesFloat(depth.data(), ucharMask.data(), 50);
381 
382  //Translate changes to the mask.
383  //If the bit is on in the mask and off in ucharMask,
384  //turn it off in the mask.
385  //Moreover, if the bit is off in the mask and on in
386  //ucharMask, it means that a hole has been filled.
387  //Turn it on in the mask, too.
388  for(int i = 0; i < ucharMask.size(); ++i)
389  {
390  if ((mask.testBit(i)) && (0 == ucharMask[i]))
391  {
392  mask.clearBit(i);
393  }
394  if ((!mask.testBit(i)) && (255 == ucharMask[i]))
395  {
396  mask.setBit(i);
397  }
398  }
399 
400  //Detrending operation.
401  //Fit a plane to the data, and then subtract it from
402  //the depth. WARNING: This modifies depth, but it
403  //is for statistical purposes.
404  qDebug() << "Removing plane slope....";
405  ManipulatePlate maniPlate(plate, depth.data());
406 
407  //Compute the coordinate system.
408  qDebug() << "Computing coordinate system....";
409  //Center at centroid.
410  QVector3D centroid;
411  for (int i = 0; i < height; ++i)
412  {
413  for (int j = 0; j < width; ++j)
414  {
415  int id = i*width + j;
416  if (mask.testBit(id))
417  {
418  centroid += QVector3D(
419  j*pixelSizeX,
420  i*pixelSizeY,
421  depth[id]);
422  }
423  }
424  }
425  //Number added in was number of "on" points.
426  centroid /= mask.count(true);
427  coordinateSystem.translate(-centroid);
428 
429  return new RangeImage(width, height, pixelSizeX, pixelSizeY,
430  depth, texture, mask, coordinateSystem);
431 }
