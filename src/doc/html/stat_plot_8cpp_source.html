91 statPlot::setProfile(Profile* profile, bool masked)
92 {
93  if ((NULL == profile) || (profile->isNull()))
94  {
95  setProfileCurve(QVector<QPointF> ());
96  setSearchWindow(0, 0);
97  emit profileChanged(NULL, true);
98  }
99  else
100  {
101  QVector<float> depth = profile->getDepth();
102  QBitArray mask = profile->getMask();
103  QVector<QPointF> profilePts;
104  float res = profile->getPixelSize();
105 
106  for (int i = 0; i < depth.size(); ++i)
107  {
108  if (masked)
109  {
110  if (mask.testBit(i))
111  profilePts.push_back(QPointF(i*res, depth[i]));
112  }
113  else
114  profilePts.push_back(QPointF(i*res, depth[i]));
115  }
116 
117  setProfileCurve(profilePts);
118  emit profileChanged(profile, masked);
119  }

123 statPlot::setSearchWindow(int searchLoc, int searchWidth)
124 {
125  if (profileCurve->boundingRect().isValid() &&
126  (searchWidth > 0)) //If profileCurve has data.
127  {
128  //Make and plot box.
129 
130  //Ensure window is not out of bounds.
131  int n = profileCurve->data()->size();
132  int startSample;
133  int endSample;
134  if (searchLoc < 0)
135  {
136  startSample = 0;
137  }
138  else if (searchLoc >= n)
139  {
140  startSample = n - 1;
141  }
142  else
143  {
144  startSample = searchLoc;
145  }
146 
147  if ((searchLoc + searchWidth) > n)
148  {
149  endSample = n - 1;
150  }
151  else
152  {
153  endSample = searchLoc + searchWidth - 1;
154  }
155 
156  QPointF tempLowerLeft
157  (profileCurve->data()->sample(startSample).x(), profileCurve->minYValue());
158  QPointF tempLowerRight
159  (profileCurve->data()->sample(endSample).x(), profileCurve->minYValue());
160  QPointF tempUpperRight = tempLowerRight;
161  QPointF tempUpperLeft = tempLowerLeft;
162  tempUpperRight.setY(profileCurve->maxYValue());
163  tempUpperLeft.setY(profileCurve->maxYValue());
164  QVector<QPointF> boxPts;
165  boxPts.push_back(tempLowerLeft);
166  boxPts.push_back(tempLowerRight);
167  boxPts.push_back(tempUpperRight);
168  boxPts.push_back(tempUpperLeft);
169  boxPts.push_back(tempLowerLeft);
170  searchWindow->setData(new QwtPointSeriesData(boxPts));
171  replot();
172  }
173  else
174  {
175  searchWindow->setData(new QwtPointSeriesData(QVector<QPointF> ()));
176  replot();
177  }
178 }
