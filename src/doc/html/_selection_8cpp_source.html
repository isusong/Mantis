55 void
56 Selection::drawColPlane()
57 {
58  //Make transparent plane.
59  //Disable writing into the depth buffer.
60  //This is a trick that prevents "stitching" in the
61  //highlight line. Since the depth of the plane
62  //is not written to the depth buffer (only used to
63  //draw the plane correctly), it can't interfere
64  //with the depth values for depth testing of the
65  //highlight line.
66  glDepthMask(GL_FALSE);
67  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
68  glColor4f(1.0f, 0.0f, 0.0f, 0.4f); //red
69  glBegin(GL_POLYGON);
70  glVertex3f(basis.x(), bbMin.y(),
71  multiplier*bbMin.z());
72  glVertex3f(basis.x(), bbMax.y(),
73  multiplier*bbMin.z());
74  glVertex3f(basis.x(), bbMax.y(),
75  multiplier*bbMax.z());
76  glVertex3f(basis.x(), bbMin.y(),
77  multiplier*bbMax.z());
78  glEnd();
79  glBlendFunc(GL_ONE, GL_ZERO); //make non-transparent.
80  glDepthMask(GL_TRUE); //Re-enable writing into the depth buffer.
81 
82  //Make highlighting line around plane.
83  glColor3f(1.0f, 0.0f, 0.0f); //red.
84  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
85  GLfloat currentLineWidth;
86  glGetFloatv(GL_LINE_WIDTH, &currentLineWidth);
87  glLineWidth(3.0f);
88  glBegin(GL_POLYGON);
89  glVertex3f(basis.x(), bbMin.y(),
90  multiplier*bbMin.z());
91  glVertex3f(basis.x(), bbMax.y(),
92  multiplier*bbMin.z());
93  glVertex3f(basis.x(), bbMax.y(),
94  multiplier*bbMax.z());
95  glVertex3f(basis.x(), bbMin.y(),
96  multiplier*bbMax.z());
97  glEnd();
98  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
99  glLineWidth(currentLineWidth);
100 }

165 void
166 Selection::redrawSelection(float x, float y)
167 {
168  float basisX = basis.x();
169  float basisY = basis.x();
170  float xdiff = fabs(basisX - x);
171  float ydiff = fabs(basisY - y);
172 
173  //This is verbosely coded to make it clear to read.
174  if ((colPlane == mode) && (xdiff > EPS))
175  {
176  updateSelectionBasis(x, y);
177  emit updateGL();
178  }
179  else if ((rowPlane == mode) && (ydiff > EPS))
180  {
181  updateSelectionBasis(x, y);
182  emit updateGL();
183  }
184 }
