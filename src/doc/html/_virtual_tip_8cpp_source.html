134 void
135 VirtualTip::computeBoundingBox()
136 {
137  //Cache some things.
138  int width = tip->getWidth();
139  int height = tip->getHeight();
140  float pixelSizeX = tip->getPixelSizeX();
141  float pixelSizeY = tip->getPixelSizeY();
142 
143  //Initialize.
144  float minX = FLT_MAX;
145  float maxX = -FLT_MAX;
146  float minY = FLT_MAX;
147  float maxY = -FLT_MAX;
148  float minZ = FLT_MAX;
149  float maxZ = -FLT_MAX;
150 
151  for (int i = 0; i < width*height; ++i)
152  {
153  if (mask.testBit(i))
154  {
155  float currentX = (i%width)*pixelSizeX;
156  float currentY = (i/width)*pixelSizeY;
157  float currentZ = depth[i];
158  if (currentX < minX)
159  minX = currentX;
160  if (currentX > maxX)
161  maxX = currentX;
162  if (currentY < minY)
163  minY = currentY;
164  if (currentY > maxY)
165  maxY = currentY;
166  if (currentZ < minZ)
167  minZ = currentZ;
168  if (currentZ > maxZ)
169  maxZ = currentZ;
170  }
171  }
172  bbMin = QVector3D(minX, minY, minZ);
173  bbMax = QVector3D(maxX, maxY, maxZ);
174 }

176 void
177 VirtualTip::computeProjection(const QMatrix4x4& transform, int* rboHeight,
178  int* partitions, float* yMin, float* yDelta)
179 {
180  //Determine orientation of tip.
181 
182  //Bounding box points indexed like this:
183  //Looking the down the negative z, we
184  //first see the plane
185  // 1 0 <-(bbMax)
186  // 2 3
187  //Then we see
188  // 5 4
189  //(bbMin)-> 6 7
190  QVector<QVector3D> boundingBox (8);
192  boundingBox[0] = QVector3D(bbMax);
193  boundingBox[1] = QVector3D(bbMin.x(), bbMax.y(), bbMax.z());
194  boundingBox[2] = QVector3D(bbMin.x(), bbMin.y(), bbMax.z());
195  boundingBox[3] = QVector3D(bbMax.x(), bbMin.y(), bbMax.z());
197  boundingBox[4] = QVector3D(bbMax.x(), bbMax.y(), bbMin.z());
198  boundingBox[5] = QVector3D(bbMin.x(), bbMax.y(), bbMin.z());
199  boundingBox[6] = QVector3D(bbMin);
200  boundingBox[7] = QVector3D(bbMax.x(), bbMin.y(), bbMin.z());
201 
202  //Manipulate the bounding box to find
203  //global x and y maxes and mins.
204  QMatrix4x4 bbTransform =
205  transform *
206  tip->getCoordinateSystemMatrix();
207  float minY = FLT_MAX;
208  float maxY = -FLT_MAX;
209  for (int i = 0; i < 8; ++i)
210  {
211  boundingBox[i] = bbTransform.map(
212  boundingBox[i]);
213 
214  if (minY > boundingBox[i].y())
215  minY = boundingBox[i].y();
216  if (maxY < boundingBox[i].y())
217  maxY = boundingBox[i].y();
218  }
219  *yMin = minY; //Store ymin.
220 
221  //Auto calculate number of partitions.
222  bool openGLError = true;
223  *partitions = 1;
224  while (openGLError && (*partitions < 10))
225  {
226  //Compute rboHeight for the correct resolution.
227  //We add one to round up to the nearest pixel.
228  *rboHeight = (maxY - minY)/((*partitions)*resolution) + 1;
229  //Compute yDelta to yield the correct resolution.
230  *yDelta = (*rboHeight)*resolution;
231 
232  //Set up "renderbuffer"
233  //I was using a renderbuffer before to hold the depth,
234  //but certain OpenGL implementations (ATI, for example)
235  //make the assumption that all framebuffers will have at
236  //least one texture object attached to them.
237  //Therefore, I needed to make this a texture for
238  //completeness.
239  //I referred to:
240  //developer.amd.com/media/gpu_assets/FramebufferObjects.pdf
241  glDeleteTextures(1, &rboID);
242  //Make and bind a new renderbuffer for holding the depth.
243  //Here's the gameplan: we make the renderbuffer 1 pixel
244  //wide and rboHeight tall. We then set glViewport to
245  //make the window representation of the data
246  //the same size. In the rasterization step,
247  //OpenGL will render the object
248  //into the nearest pixels available.
249  glGenTextures(1, &rboID);
250  glBindTexture(GL_TEXTURE_2D, rboID);
251  glTexImage2D(GL_TEXTURE_2D, 0,
252  GL_DEPTH_COMPONENT32F, //Floating pt. format.
253  1, *rboHeight, 0, GL_DEPTH_COMPONENT, GL_FLOAT, NULL); //Allocate it.
254  // From Nik: This checks for any openGL errors.
255  GLenum errCode;
256  if ((errCode = glGetError()) != GL_NO_ERROR)
257  {
258  cout << "OpenGL Error: " <<
259  gluErrorString(errCode) << endl;
260  (*partitions)++; //Increase the number of partitions.
261  //openGLError stays true.
262  }
263  else
264  {
265  //Break the loop.
266  openGLError = false;
267  }
268  }
269  qDebug() << "Number of partitions = " << *partitions;
270  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT,
271  GL_DEPTH_ATTACHMENT_EXT, GL_TEXTURE_2D,
272  rboID, 0); //Attach to fbo.
273 
274  //Need to tell the driver that we're not using color, only
275  //depth. Otherwise, it complains that the fbo is incomplete.
276  glDrawBuffer(GL_NONE);
277  glReadBuffer(GL_NONE);
278 
279  //Is the fbo complete?
280  GLenum completeness =
281  glCheckFramebufferStatusEXT(GL_FRAMEBUFFER_EXT);
282  if (GL_FRAMEBUFFER_COMPLETE_EXT != completeness)
283  {
284  qDebug() << "Error: Framebuffer incomplete." <<
285  "Error code: " << completeness;
286  }
287 
288  //Set viewport to appropriate size.
289  glViewport(0, 0, 1, *rboHeight);
290 }

314 void
315 VirtualTip::draw()
316 {
317  //To ease the load on OpenGL, I'm using a StreamBuffer
318  //object to actually draw the triangles.
319 
320  //Need to use unique meshing strategy here due to
321  //squish matrix. The OpenGL polygon rasterizer won't
322  //draw anything at 1 thick. See
323  //http://www.opengl.org/archives/resources/faq/technical/rasterization.htm, item 14.120.
324  //So we employ a "cat's cradle" strategy.
325  //We mesh _between_ two instances of the data, a left and a right.
326  //We set w in the point to tell the vertex shader which is which.
327  //Then the vertex shader pulls the two instances apart.
328 
329  //cache some things.
330  int width = tip->getWidth();
331  int height = tip->getHeight();
332  float pixSizeX = tip->getPixelSizeX();
333  float pixSizeY = tip->getPixelSizeY();
334  float* depthPtr = depth.data(); //don't delete this.
335 
336  sbuffer->bind();
337 
338  //Make triangles and draw them.
339  for (int i = 0; i < height - 1; ++i) //Spaces between rows.
340  {
341  for (int j = 0; j < width-1; ++j) //Spaces between cols.
342  {
343  //Zoom in a point 0 at j, i, and its right
344  //neighbor, neighbor below, and its
345  //neighbor below and to the right.
346  //Like this:
347  // 0 1
348  // 2 3
349  int idx0 = width*i + j;
350  int idx1 = idx0 + 1;
351  int idx2 = width + idx0;
352  int idx3 = idx2 + 1;
353 
354  //Need x and y coordinates for each point.
355  float x0 = j*pixSizeX;
356  float y0 = i*pixSizeY;
357  float x1 = x0 + pixSizeX;
358  float y2 = y0 + pixSizeY;
359 
360  //Can I make triangles between 0l/1l and 0r/1r?
361  if (mask.testBit(idx0) & mask.testBit(idx1))
362  {
363  makeTriangles(x0, y0, depthPtr[idx0],
364  x1, y0, depthPtr[idx1]);
365  }
366  //Can I make triangles between 1l/3l and 1r/3r?
367  if ((mask.testBit(idx1) & mask.testBit(idx3))
368  && (j == width - 1)) //only draw at end.
369  {
370  makeTriangles(x1, y0, depthPtr[idx1],
371  x1, y2, depthPtr[idx3]);
372  }
373  //Can I make triangles between 2l/3l and 2r/3r?
374  if ((mask.testBit(idx2) & mask.testBit(idx3))
375  && (i == height - 1)) //only draw at end.
376  {
377  makeTriangles(x1, y2, depthPtr[idx3],
378  x0, y2, depthPtr[idx2]);
379  }
380  //Can I make triangles between 0l/2l and 0r/2r?
381  if (mask.testBit(idx0) & mask.testBit(idx2))
382  {
383  makeTriangles(x0, y2, depthPtr[idx2],
384  x0, y0, depthPtr[idx0]);
385  }
386  //Can I make triangles between 1l/2l and 1r/2r?
387  if (mask.testBit(idx1) & mask.testBit(idx2))
388  {
389  makeTriangles(x0, y2, depthPtr[idx2],
390  x1, y0, depthPtr[idx1]);
391  }
392  //Can I make triangles between 0l/3l and 0r/3r?
393  if (mask.testBit(idx0) & mask.testBit(idx3))
394  {
395  makeTriangles(x0, y0, depthPtr[idx0],
396  x1, y2, depthPtr[idx3]);
397  }
398  }
399  }
400 
401  //Draw the leftovers, if any.
402  sbuffer->flush();
403 }

462 Profile*
463 VirtualTip::mark(float xAxis, float yAxis, float zAxis)
464 {
465  //Declare/initialize some things.
466  int rboHeight; //Number of pixels in 1D "renderbuffer."
467  int partitions; //Number of scene partitions.
468  float yMin; //Bottom edge of the overall scene.
469  float yDelta; //Length of a partition in um.
470  //Create the transform matrix.
471  QMatrix4x4 transform;
472  //Transforms in PYR order.
473  //This causes it to rotate around the fixed world
474  //coordinate system.
475  transform.rotate(zAxis, QVector3D(0, 0, 1)); //z-roll
476  transform.rotate(yAxis, QVector3D(0, 1, 0)); //y-yaw
477  transform.rotate(xAxis, QVector3D(1, 0, 0)); //x-pitch
478 
479  //Activate the context.
480  context->makeCurrent();
481  glEnable(GL_DEPTH_TEST); //so OpenGL will use a depth buffer.
482 
483  //Bind framebuffer for reading and writing.
484  //Now OpenGL draws into this
485  //instead of the screen.
486  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, fboID);
487 
488  //Create the rbo and determine the correct projection.
489  computeProjection(transform, &rboHeight, &partitions, &yMin, &yDelta);
490 
491  //Set up the modelview stack.
492  glMatrixMode(GL_MODELVIEW);
493  glLoadMatrixd(camera.constData());
494  QMatrix4x4 squish;
495  squish(0, 0) = 0;
496  glMultMatrixd(squish.constData()); //squish in x.
497  glMultMatrixd(transform.constData());
498  glMultMatrixd(tip->getCoordinateSystemMatrix().constData());
499 
500  //Bind the correct shader program.
501  prog->bind();
502 
503  //Capture each mark partition and collect them.
504  QVector<float> rawMarkData; //to store the mark.
505  float bottomClip = yMin;
506  float topClip = yMin + yDelta;
507  for (int i = 0; i < partitions; ++i)
508  {
509  //Clear fbo
510  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
511 
512  //Compute new projection matrix.
513  QMatrix4x4 projectionMatrix;
514  projectionMatrix.ortho(-0.5, 0.5,
515  bottomClip, topClip, NEARCLIP, FARCLIP);
516 
517  //Load perspective matrix into OpenGL
518  glMatrixMode(GL_PROJECTION);
519  //Overwrite the current projection matrix.
520  glLoadMatrixd(projectionMatrix.constData());
521 
522  //Draw the tip.
523  draw();
524 
525  // From Nik: This checks for any openGL errors.
526  GLenum errCode;
527  if ((errCode = glGetError()) != GL_NO_ERROR)
528  {
529  cout << "OpenGL Error: " <<
530  gluErrorString(errCode) << endl;
531  }
532 
533  //Pull depth out of the framebuffer object.
534  GLfloat* data = new GLfloat [rboHeight];
535  glReadPixels(0, 0, 1, rboHeight,
536  GL_DEPTH_COMPONENT, GL_FLOAT, data);
537  //stash it.
538  for (int j = 0; j < rboHeight; ++j)
539  {
540  rawMarkData.push_back(data[j]);
541  }
542  delete [] data;
543 
544  //Update the clip planes for the next run.
545  bottomClip += yDelta;
546  topClip += yDelta;
547  }
548 
549  //Remove the masked data from the ends.
550  float maskValue;
551  glGetFloatv(GL_DEPTH_CLEAR_VALUE, &maskValue);
552  QVector<float> markData =
553  CleanVirtualMark::unmask(rawMarkData, maskValue);
554 
555  //Convert data to um.
556  CleanVirtualMark::depthToUm(markData, NEARCLIP,
557  FARCLIP, CAMERAZ);
558 
559  //Flip the mark data up/down.
560  //This needs to happen because the mark is an impression.
561  for (int i = 0; i < markData.size(); ++i)
562  {
563  markData[i] = -markData[i];
564  }
565  //Flip the mark left/right.
566  QVector<float> markDataCopy (markData);
567  int markDataSize = markData.size();
568  for (int i = 0; i < markDataSize; ++i)
569  {
570  markData[i] = markDataCopy[markDataSize - i - 1];
571  }
572 
573  //Find the mark edges for trimming.
574  QPoint edges = CleanVirtualMark::findMarkEdges(markData);
575 
576  //Make Profile.
577  QBitArray profileMask (markDataSize, true);
578  for (int i = 0; i < edges.x(); ++i)
579  profileMask.clearBit(i);
580  for (int i = edges.y() + 1; i < markDataSize; ++i)
581  profileMask.clearBit(i);
582  Profile* ret = new Profile(resolution, markData, profileMask);
583 
584  //Clean up.
585  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, 0);
586  return ret;
587 }
