77 QVector<float>
78 StatInterface::trimProfileEnds(Profile* data)
79 {
80  //Cache some things.
81  QVector<float> depth = data->getDepth();
82  QBitArray mask = data->getMask();
83 
84  //Find the ends to trim off.
85  int dataLength = depth.size();
86  int startIdx, endIdx;
87  for (int i = 0; i < dataLength; ++i)
88  {
89  if (mask.testBit(i))
90  {
91  startIdx = i;
92  break;
93  }
94  }
95 
96  for (int i = 0; i < dataLength; ++i)
97  {
98  if (mask.testBit(dataLength - 1 - i))
99  {
100  endIdx = dataLength - 1 - i;
101  break;
102  }
103  }
104 
105  //Trim it.
106  QVector<float> ret;
107  for (int i = startIdx; i < endIdx + 1; ++i)
108  ret.push_back(depth[i]);
109 
110  return ret;
111 }

113 void
114 StatInterface::compare(QVector<float> data1, QVector<float> data2)
115 {
116  //Vector prep.
117  ConvertTraceToInt intConverter;
118  auto_ptr<vector<int> > trace1 =
119  intConverter(toDouble(data1).toStdVector());
120  auto_ptr<vector<int> > trace2 =
121  intConverter(toDouble(data2).toStdVector());
122 
123  //Run without checking for flips.
124  //maxCorWithFlips is in flipcorrelation.h.
125  //Ru He says, "Once searchWindow is fixed, the
126  //returned c.loc1(), c.loc2() in c object should be unique."
127  const int length1 = trace1->size();
128  const int length2 = trace2->size();
129  FlippableCorLoc c (0, 0, 0, true); //"default" FCL object.
130  try
131  {
132  c = maxCorWithFlips(trace1->begin(),
133  trace2->begin(),
134  length1,
135  length2,
136  searchWindow,
137  maxShiftPercentage,
138  false);
139  } catch (runtime_error err) {
140  qDebug() << "There was a runtime error in the" <<
141  "stat package:";
142  qDebug() << err.what();
143  throw err; //Throw it up to the next level.
144  return; //exit from this function. r and t and loc1 and loc2 will stale.
145  }
146 
147  //Calculate T value (and average if T_sample_size is > 1).
148  QVector<double> T_vector;
149  for (int i = 0; i < T_sample_size; ++i)
150  {
151  //Maverick says, "[R]igid pairs correlation.
152  //In other words, the correlation for two windows
153  //that has [sic] the same shifts with respect to the
154  //position of maximum correlation."
155  //IntRigidCorSampExcludeSearch is in intnolev_functors.h
156  IntRigidCorSampExcludeSearch compRigidCor;
157  auto_ptr<vector<double> > rigidCor =
158  compRigidCor(*trace1, *trace2, c.loc1(), c.loc2(),
159  searchWindow, numRigidPairs, validWindow);
160  if (0 == rigidCor->size())
161  {
162  //Ran out of space for the validation window.
163  QString what (tr("The rigid-shift "
164  "validation window "
165  "did not fit inside one of the trace data sets. "
166  "Bear in mind that the validation window "
167  "cannot lie inside the search window.\n\n"
168  "Try a smaller validation and/or search window."));
169  qDebug() << what;
170  throw std::range_error(what.toStdString());
171  return; //exit from this function.
172  }
173 
174  //Maverick says, "For the random pairs correlation,
175  //the shifts of two pairs are different w.r.t. to [sic]
176  //the position of maximum correlation.
177  //IntRandomCorSampExcludeSearch is in intnolev_functors.h
178  IntRandomCorSampExcludeSearch compRandomCor;
179  auto_ptr<vector<double> > randomCor =
180  compRandomCor(*trace1, *trace2, c.loc1(), c.loc2(),
181  searchWindow, numRandomPairs, validWindow);
182  if (0 == randomCor->size())
183  {
184  //Ran out of space for the validation window.
185  QString whatnow (tr("The random-shift "
186  "validation window "
187  "did not fit inside one of the trace data sets. "
188  "Bear in mind that the validation window "
189  "cannot lie inside the search window.\n\n"
190  "Try a smaller validation and/or search window."));
191  qDebug() << whatnow;
192  throw std::range_error(whatnow.toStdString());
193  return; //exit from this function.
194  }
195 
196  //Compute T value
197  T_vector.push_back(t1Statistic(*rigidCor, *randomCor));
198  }
199  //Average T values.
200  //If we made it this far, there were no errors.
201  double T_mean, T_var;
202  meanAndVar(T_vector.toStdVector(), T_mean, T_var);
203 
204  //Store the results.
205  tValue = T_mean;
206  rValue = c.cor();
207  loc1 = c.loc1();
208  loc2 = c.loc2();
209 }
