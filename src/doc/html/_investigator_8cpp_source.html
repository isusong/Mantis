108 void
109 Investigator::openTip()
110 {
111  //Get the filename.
112  QString filename = QFileDialog::getOpenFileName(this,
113  tr("Open .mt file"), lastDirectory,
114  tr("Mantis File (*.mt)"));
115  if (filename.isNull()) return; //user canceled.
116 
117  //Update the last directory.
118  fileInfo.setFile(filename);
119  lastDirectory = fileInfo.absolutePath();
120 
121  //Create the new widget.
122  QMdiSubWindow* window = new QMdiSubWindow(area);
123  window->setAttribute(Qt::WA_DeleteOnClose);
124  RangeImage* tip = new RangeImage(filename, window);
125  if (tip->isNull())
126  {
127  //Deletes the window and the tip object.
128  window->close();
129  QMessageBox::warning(this, tr("Error"),
130  tr("File failed to open."));
131  return;
132  }
133  TipWidget* tipWidget = new TipWidget(tip, window);
134  window->setWidget(tipWidget);
135  window->setWindowTitle(fileInfo.fileName());
136  area->addSubWindow(window);
137  tipWidget->show();
138  area->tileSubWindows();
139  assignSlot(tipWidget);
140 }

205 Investigator::updateEnabledStatus()
206 {

208  //unless you close a tip or a plate and free up the slot.
209  if (slotFilled_1 && slotFilled_2)

219

222  {
223  tileAction->setEnabled(true);
