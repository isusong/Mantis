249 RangeImage::saveDepth(const QString& fname) const
250 {
251  QImage graydepth (width, height, QImage::Format_Indexed8);
252  graydepth.setColorCount(256);
253  for (int i = 0; i < 256; ++i)
254  graydepth.setColor(i, qRgb(i, i, i));
255  graydepth.fill(0);
256 
257  //Find the depth range.
258  float minZ = FLT_MAX;
259  float maxZ = -FLT_MAX;
260  for (int i = 0; i < width*height; ++i)
261  {
262  if (mask.testBit(i))
263  {
264  float currentZ = depth[i];
265  if (currentZ < minZ)
266  minZ = currentZ;
267  if (currentZ > maxZ)
268  maxZ = currentZ;
269  }
270  }
271 
272  //Turn on the appropriate white pixels.
273  for (int i = 0; i < height; ++i)
274  {
275  for (int j = 0; j < width; ++j)
276  {
277  if (mask.testBit(width*i + j))
278  {
279  int colorIdx =
280  floor(255*((depth[width*i + j] - minZ)/(maxZ - minZ)));
281  graydepth.setPixel(j, i, colorIdx);
282  }
283  }
284  }
285 
286  return graydepth.save(fname);

364 RangeImage*
365 RangeImage::importFromTxyz(const QString& fname)
366 {
367  //Report status
368  QString status ("Importing ");
369  status.append(fname);
370  status.append(" from Txyz");
371  qDebug() << status;
372  //emit statusMessage(status);
373 
374  //This is code inspired by Song Zhang's TxyzFileIO class.
375  FILE* fp;
376  fp = fopen(fname.toStdString().c_str(), "rb");
377  if (NULL == fp) //If filestream is bad
378  {
379  QString warning ("Error opening file ");
380  warning.append(fname);
381  qDebug() << warning;
382  //emit warningMessage(warning);
383  return NULL;
384  }
385 
386  //Parameters that are needed from the file.
387  int width, height;
388  float pixelSize;
389  QVector<float> depth;
390  QImage texture;
391  QBitArray mask;
392  QMatrix4x4 coordinateSystem;
393 
394  //Grab size of file and pixel size.
395  fscanf(fp, "image size width x height x pixelSize = %d x %d x %f",
396  &width, &height, &pixelSize);
397  int imageSize = width * height;
398 
399  //Read bw texture data and convert to QImage.
400  unsigned char * bwTexture = new unsigned char [imageSize];
401  fread(bwTexture, sizeof(unsigned char), imageSize, fp);
402  QImage temp = QImage (bwTexture, width, height,
403  sizeof(unsigned char) * width, QImage::Format_Indexed8);
404  //Need to copy here because temp is only valid as long
405  //as bwTexture exists, and I want to delete bwTexture soon.
406  texture = temp.copy();
407  //Set the color table to B/W.
408  texture.setColorCount(256);
409  for (int i = 0; i < 256; ++i) //set the color table.
410  texture.setColor(i, qRgb(i, i, i));
411 
412  //Read depth data.
413  float * depthData = new float [imageSize];
414  fread(depthData, sizeof(float), imageSize, fp);
415  for (int i = 0; i < imageSize; ++i)
416  depth.push_back(depthData[i]);
417  delete [] depthData;
418 
419  //Interpret the mask from the texture.
420  mask.fill(false, imageSize);
421  for (int i = 0; i < imageSize; ++i)
422  {
423  if (bwTexture[i]) //If the texture is nonzero.
424  mask.setBit(i); //Set the mask there to true.
425  }
426  delete [] bwTexture;
427 
428  //Clean up.
429  fclose(fp);
430 
431  return new RangeImage(width, height, pixelSize, pixelSize,
432  depth, texture, mask, coordinateSystem);
433 }

436 RangeImage::importFromAl3d(const QString& fname,
437  const QString& texfname)
438 {
439  //Report status.
440  QString status ("Importing ");
441  status.append(fname);
442  status.append(" from al3d.");
443  qDebug() << status;
444  //emit statusMessage(status);
445 
446  //Parameters that are needed from the file.
447  int width, height;
448  float pixelSizeX, pixelSizeY;
449  QVector<float> depth;
450  QImage texture;
451  QBitArray mask;
452  QMatrix4x4 coordinateSystem;
453 
454  //Create an al3d object for the file.
455  AL3DFile al3d (fname);
456  if (al3d.isNull())
457  {
458  QString warning ("Error importing al3d file ");
459  warning.append(fname);
460  qDebug() << warning;
461  //emit warningMessage(warning);
462  return NULL;
463  }
464 
465  //Get width, height, pixelSize.
466  width = al3d.getWidth();
467  height = al3d.getHeight();
468  pixelSizeX = al3d.getPixelSizeX()*CONVERT;
469  pixelSizeY = al3d.getPixelSizeY()*CONVERT;
470  if (pixelSizeX != pixelSizeY)
471  {
472  qDebug() << fname << ": pixelSizeX != pixelSizeY.";
473  }
474 
475  //Get depth.
476  //al3d object owns the depthData pointer and deletes it.
477  const float * depthData = al3d.getDepth();
478  if (!depthData)
479  {
480  QString warning (fname);
481  warning.append(" does not contain depth data.");
482  qDebug() << warning;
483  //emit warningMessage(warning);
484  return NULL;
485  }
486  for (int i = 0; i < width*height; ++i)
487  depth.push_back(depthData[i]*CONVERT);
488 
489  //Get texture
490  if (texfname.isEmpty()) //I haven't specified an external texture.
491  {
492  //Grab the texture from the file.
493  if (!al3d.get_image_by_tag("TexturePtr").isNull())
494  texture = al3d.get_image_by_tag("TexturePtr");
495  else //error, no texture info.
496  {
497  QString warning (fname);
498  warning.append(" does contain 2D texture");
499  warning.append(" and no additional texture file was specified.");
500  qDebug() << warning;
501  //emit warningMessage(warning);
502  return NULL;
503  }
504  }
505  else
506  {
507  //Load texture from external file.
508  if (!texture.load(texfname))
509  {
510  QString msg (texfname);
511  msg.append(": Invalid file or filename.");
512  qDebug() << msg;
513  //emit warningMessage(msg);
514  return NULL;
515  }
516  }
517 
518  //Get mask.
519  mask = al3d.getMask();
520  if (mask.isEmpty())
521  mask = QBitArray(width*height, true);
522 
523  //Success
524  return new RangeImage(width, height, pixelSizeX, pixelSizeY,
525  depth, texture, mask, coordinateSystem);
526 }

528 bool
529 RangeImage::exportToOBJ(const QString& fname)
530 {
531  //Report status.
532  QString status ("Exporting to obj file ");
533  status.append(fname);
534  qDebug() << status;
535 
536  //Open the file.
537  QFile file (fname);
538  if (!file.open(QIODevice::WriteOnly))
539  {
540  QString warning ("Error opening file ");
541  warning.append(fname);
542  qDebug() << warning;
543  return false;
544  }
545 
546  //Do meshing (populate indices).
547  QVector<int> indices;
548  //Go through the spaces between the rows.
549  for (int i = 0; i < height - 1; ++i)
550  {
551  //Go through the spaces between the columns.
552  for (int j = 0; j < width - 1; ++j)
553  {
554  //Zoom in on a point 0 at i, j and its
555  //neighbor on the right, point 1, its
556  //neighbor below, point 2, and its
557  //neighbor below and to the right, point 3.
558  //Like this:
559  // 0 1
560  // 2 3
561  int index0 = width*i + j;
562  int index1 = index0 + 1;
563  int index2 = index0 + width;
564  int index3 = index2 + 1;
565 
566  //Can I connect point 1 to point 2?
567  if (mask.testBit(index1) & mask.testBit(index2))
568  {
569  // I use the default / method to divide the
570  //triangles.
571  if(mask.testBit(index0)) //Can I connect 0 to 1 and 2?
572  {
573  //If so, I can make a triangle out of them.
574  indices.push_back(index0);
575  indices.push_back(index1);
576  indices.push_back(index2);
577  }
578  if(mask.testBit(index3)) //Can I connect 3 to 1 and 2?
579  {
580  indices.push_back(index2);
581  indices.push_back(index3);
582  indices.push_back(index1);
583  }
584  }
585  else if (mask.testBit(index0) & mask.testBit(index3)) //Can I connect 0 and 3?
586  {
587  //I use the \ method the divide the triangles.
588  if(mask.testBit(index1)) //Can I connect 1 to 0 and 3?
589  {
590  indices.push_back(index1);
591  indices.push_back(index3);
592  indices.push_back(index0);
593  }
594  if(mask.testBit(index2)) //Can I connect 2 to 0 and 3?
595  {
596  indices.push_back(index2);
597  indices.push_back(index3);
598  indices.push_back(index0);
599  }
600  }
601  }
602  }
603 
604  //Serializer object for QT classes.
605  QTextStream fileWriter(&file);
606  //File body.
607  //vertices and texture coordinates
608  for (int i = 0; i < height; ++i)
609  {
610  for (int j = 0; j < width; ++j)
611  {
612  fileWriter << "v "
613  << j*pixelSizeX << " "
614  << i*pixelSizeY << " "
615  << depth[width*i + j] << " "
616  << "\n";
617  fileWriter << "vt "
618  << ((1/((float) width)) * (j + 0.5f)) //s
619  << " "
620  << ((1/((float) height))* (i + 0.5f)) //t
621  << " \n";
622  }
623  }
624  //faces
625  int ntri = indices.size()/3;
626  for (int i = 0; i < ntri; ++i)
627  {
628  int index1 = indices[(3*i) + 0] + 1;
629  int index2 = indices[(3*i) + 1] + 1;
630  int index3 = indices[(3*i) + 2] + 1;
631  fileWriter << "f " << index1 << " "
632  << index2 << " "
633  << index3 << " \n";
634  }
635 
636  file.close();
637 
638  return true;
639 }

643 {
644  int dWidth = width/skip;
645  int dHeight = height/skip;
646  QVector<float> dDepth;
647  QImage dTexture (dWidth, dHeight, texture.format());
648  QBitArray dMask (dWidth*dHeight, true);

663  skip*pixelSizeY, dDepth, dTexture, dMask,
