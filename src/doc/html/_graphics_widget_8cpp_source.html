79 void
80 GraphicsWidget::paintGL()
81 {
82  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
83  if (model)
84  {
85  glMatrixMode(GL_MODELVIEW);
86  glLoadMatrixd(sceneTranslation.constData());
87  glMultMatrixd(sceneRotation.constData());
88 
89  model->draw(this);
90 
91  //Draw world coordinate system.
92  //The model is in charge of detaching its shader programs, if any.
93  if (drawCS)
94  {
95  glDisable(GL_DEPTH_TEST);
96  glEnable(GL_LINE_STIPPLE); //dashed line
97  float lineLength = 4e3;
98  glBegin(GL_LINES);
99  glColor3f(0.6f, 0, 0); //red pen
100  glVertex3f(0, 0, 0);
101  glVertex3f(lineLength, 0, 0); //x
102  glColor3f(0, 0.6f, 0); //green pen
103  glVertex3f(0, 0, 0);
104  glVertex3f(0, lineLength, 0); //y
105  glColor3f(0, 0, 0.6f); //blue pen
106  glVertex3f(0, 0, 0);
107  glVertex3f(0, 0, lineLength);
108  glEnd();
109  glDisable(GL_LINE_STIPPLE);
110  glEnable(GL_DEPTH_TEST);
111  }
112  }

158 void
159 GraphicsWidget::setModel(GenericModel* newModel)
160 {
161  //QPointer now points to the same thing as newModel.
162  model = newModel;
163 
164  //Cache bounding box.
165  QVector3D bbExtent = model->getBbMax() - model->getBbMin();
166  bbX = bbExtent.x();
167  bbY = bbExtent.y();
168  bbZ = bbExtent.z();
169 
170  //Set up camera.
171  sceneRotation = QMatrix4x4();
172  sceneTranslation = QMatrix4x4();
173  zRecommendedView = 1.5*bbY/(2*qTan(FOVY/2));
174  sceneTranslation.translate(0, 0, -zRecommendedView);
175 }

185 void
186 GraphicsWidget::mouseMoveEvent(QMouseEvent* event)
187 {
188  if (!model) return;
189  if (Qt::LeftButton & event->buttons())
190  {
191  QVector3D delta = QVector3D(capturedPos - event->pos());
192  delta.setX(-delta.x()); //Correct for window coordinates.
193  //Convert to world coordinates -ish.
194  delta *= (mouseMult.y()*bbY)/height();
195  //Correct for zoom -ish.
196  delta *= qAbs(sceneTranslation(2, 3))/zRecommendedView;
197  sceneTranslation.translate(delta);
198  }
199  if (Qt::RightButton & event->buttons())
200  {
201  sphere.move(screenToNDC(event->pos()), quat);
202  }
203  capturedPos = event->pos();
204  updateGL();
205 }
